 59     rgbToHsl: function(R,G,B)
 60    	{ 
 61 		var pThis = Eco.Color;
 62 		var HSLMAX = pThis.HSLMAX;
 63 		var RGBMAX = pThis.RGBMAX;
 64 		var cMax = Math.max( Math.max(R,G), B);
 65 		var cMin = Math.min( Math.min(R,G), B);
 66 		var L = Math.floor(( ((cMax+cMin)*HSLMAX) + RGBMAX )/(2*RGBMAX) );
 67 		var H = 0;
 68 		
 69 		if (cMax == cMin) 
 70 		{                               /* r=g=b --> achromatic case */
 71 			 S = 0;                     /* saturation */
 72 			 H = pThis.HUE_VALUE_WHEN_SATURATION_IS_ZERO;        /* hue */
 73 			 
 74 			 //console.log("pThis.HUE_VALUE_WHEN_SATURATION_IS_ZERO=" +pThis.HUE_VALUE_WHEN_SATURATION_IS_ZERO)
 75 			 //console.log("cMax="+ cMax + ", cMin="+ cMin  + ", S="+ S + ", H="+ H);	
 76 		}
 77 		else 
 78 		{                        /* chromatic case */
 79 			 /* saturation */
 80 			 if (L <= (HSLMAX/2))
 81 			 {
 82 				S = Math.floor( ( ((cMax-cMin)*HSLMAX) + ((cMax+cMin)/2) ) / (cMax+cMin) );
 83 			 }
 84 			 else
 85 			 {
 86 				S = Math.floor( ( ((cMax-cMin)*HSLMAX) + ((2*RGBMAX-cMax-cMin)/2) )/ (2*RGBMAX-cMax-cMin) );
 87 			 }
 88 
 89 				 /* hue */
 90 			  var Rdelta = Math.floor( ( ((cMax-R)*(HSLMAX/6)) + ((cMax-cMin)/2) ) / (cMax-cMin) );
 91 			  var Gdelta = Math.floor( ( ((cMax-G)*(HSLMAX/6)) + ((cMax-cMin)/2) ) / (cMax-cMin) );
 92 			  var Bdelta = Math.floor( ( ((cMax-B)*(HSLMAX/6)) + ((cMax-cMin)/2) ) / (cMax-cMin) );
 93 
 94 			 if (R == cMax)
 95 			 {
 96 				H = Bdelta - Gdelta;
 97 			 }
 98 			 else if (G == cMax)
 99 			 {
100 				H = (HSLMAX/3) + Rdelta - Bdelta;
101 			 }
102 			 else /* B == cMax */
103 			 {
104 				H = ((2*HSLMAX)/3) + Gdelta - Rdelta;
105 			 }
106 			 
107 			 if (H < 0)	H += HSLMAX;
108 			 
109 			 if (H > HSLMAX) H -= HSLMAX;
110 		}
111 
112 		var res = [];
113 		res[0] = Math.floor(H);
114 		res[1] = Math.floor(S);
115 		res[2] = Math.floor(L);
116 		return res;
117 	},

120 	/**
121 	 * 색상을 RGB로 변환한다.
122 	 * @private 
123 	 * @param {number} n1 보정값1.
124 	 * @param {number} n2 보정값2.
125 	 * @param {number} hue 색상.
126 	 * @return {number} 색상.
127 	 * @memberOf Eco.Color
128 	 */     
129    	hueToRgb: function(n1,n2,hue)
130    	{
131 	   var HSLMAX = Eco.Color.HSLMAX;
132 	   
133 		n1 = Math.floor(n1);
134 		n2 = Math.floor(n2);
135 		hue = Math.floor(hue);
136       /* range check: note values passed add/subtract thirds of range */
137       if (hue < 0) hue += HSLMAX;
138 
139       if (hue > HSLMAX) hue -= HSLMAX;
140 
141       /* return r,g, or b value from this tridrant */
142       if (hue < (HSLMAX/6)) 
143       {
144           return Math.floor( n1 + Math.floor((((n2-n1)*hue+(HSLMAX/12))/(HSLMAX/6))) );
145       }
146       
147       if (hue < (HSLMAX/2))
148       {
149          return ( n2 );
150       }
151       
152       
153       if (hue < ((HSLMAX*2)/3))
154       {
155          return Math.floor( n1 +    Math.floor( (((n2-n1)*(((HSLMAX*2)/3)-hue)+(HSLMAX/12)) / (HSLMAX/6)) ) );
156       }
157       else
158       {
159          return Math.floor( n1 );
160       }
161

165 	/**
166 	 * HSL을 RGB로 변환한다.
167 	 * @param {number} hue 색상.
168 	 * @param {number} sat 채도.
169 	 * @param {number} lum 명도.
170 	 * @return {array.<number>} RGB값을 가진 array. [R,G,B].
171 	 * @example 
172 	 * trace(Eco.Color.hslToRgb(100,240,120)); //output: [0,255,128]
173 	 * trace(Eco.Color.hslToRgb(20,240,120));  //output: [255,128,0]
174 	 *
175 	 * @memberOf Eco.Color
176 	 */     
177    	hslToRgb: function(hue,sat,lum)
178    	{
179 	  var R,G,B;
180 	  var pThis = Eco.Color;
181 	  var HSLMAX = pThis.HSLMAX;
182 	  var RGBMAX = pThis.RGBMAX;
183 
184 	 //trace("\t### Eco.hlsToRGB > hue=" + hue + ", sat=" + sat + ", lum=" + lum);
185       if (sat == 0) 
186       {  /* achromatic case */
187          R=G=B=(lum*RGBMAX)/HSLMAX;
188         
189          if (hue != pThis.HUE_VALUE_WHEN_SATURATION_IS_ZERO) 
190          {
191             /* ERROR */
192          }
193       }
194       else  
195       {  
196          //보정처리
197          /* chromatic case */
198          /* set up magic numbers */
199          var Magic1, Magic2;
200          
201          if (lum <= (HSLMAX/2))
202          {
203 			Magic2 = Math.floor(	(lum*(HSLMAX + sat) + (HSLMAX/2)) / HSLMAX	);
204          }
205          else
206          {
207             Magic2 = (lum + sat - Math.floor( ((lum*sat) + (HSLMAX/2))/HSLMAX) );
208          }
209 			
210          Magic1 = 2*lum-Magic2;
211 
212 		 var hueToRgb= pThis.hueToRgb;
213          /* get RGB, change units from HSLMAX to RGBMAX */
214          R = ( hueToRgb(Magic1,Magic2,hue+(HSLMAX/3)) * RGBMAX + (HSLMAX/2) )  / HSLMAX;
215          G = ( hueToRgb(Magic1,Magic2,hue) * RGBMAX + (HSLMAX/2) )  / HSLMAX;
216          B = ( hueToRgb(Magic1,Magic2,hue - (HSLMAX/3)) * RGBMAX +  (HSLMAX/2) )  / HSLMAX;
217       }
218 	  
219 	  res = [];
220 	  res[0] = Math.floor(R);
221 	  res[1] = Math.floor(G);
222 	  res[2] = Math.floor(B);
223 	  
224 	  return res;
225     },

285 	 * Hexadecimal code를 [r, g, b, a]로 변환한다.
286 	 * @param {string} str "red"같이 named color나, "#000000", "#000000ff" 값들이 주어진다.
287 	 * @return {array} [r, g, b, a] 형태의 array 값.
288 	 * @example
289 	 * trace(Eco.Color.hexToRgb("#FF8C00")); //output: [255,140,0]
290 	 * @memberOf Eco.Color
291 	 */	
292 	hexToRgb: function(str)
293 	{
294 		if(Eco.isEmpty(str)) {
295 			alert("Arguments is empty!");
296 			return;
297 		}
298 		
299 		if ( !(str.match(/^#[0-9a-f]{3,8}$/i)) )
300 		{
301 			str = nexacro._xreNamedColorList[str];
302 		}
303 		
304 		var hex;
305 		if ( str.length == 9 )
306 		{
307 			hex = str.match(/^#?(\w{1,2})(\w{1,2})(\w{1,2})(\w{1,2})$/);
308 		}
309 		else
310 		{
311 			hex = str.match(/^#?(\w{1,2})(\w{1,2})(\w{1,2})$/);
312 		}
313 		if (hex.length >= 4)
314 		{
315 			var rgb = [0, 0, 0];
316 			for (var i = 0; i < 3; i++)
317 			{
318 				var value = hex[i + 1];
319 				rgb[i] = parseInt(value.length == 1
320 						? value + value : value, 16) ;
321 			}
322 			if ( hex[4] && hex[4].length > 0 )
323 			{
324 				rgb.push(parseInt(hex[4], 16) );
325 			}
326 			else
327 			{
328 				rgb.push(255);			
329 			}
330 			return rgb;
331 		}
