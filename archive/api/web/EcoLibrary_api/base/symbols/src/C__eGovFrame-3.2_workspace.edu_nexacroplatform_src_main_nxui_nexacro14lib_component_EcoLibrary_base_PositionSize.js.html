 69 				
 70 				var pThis = Eco.XComp.PositionSize;
 71 				
 72 				x = system.screenToClientX(XCompA, tempX) + pThis.getScrollLeft(XCompA);
 73 				y = system.screenToClientY(XCompA, tempY) + pThis.getScrollTop(XCompA);				
 74 			}
 75 			else
 76 			{
 77 				x = xy[0];
 78 				y = xy[1];
 79 			}
 80 			
 81 			return [x, y];

369 		 *
370 		 * @memberOf Eco.XComp.PositionSize
371 		 */	
372 		getTextSize: function(XComp, text, multiline, content_width, fitText)
373 		{
374 			var font;
375 			if ( XComp instanceof nexacro.Style_font )
376 			{
377 				font = XComp;
378 				multiline = Eco.isEmpty(multiline) ? false : multiline;
379 				if ( multiline && content_width == null )
380 				{
381 					content_width = 50;
382 				}
383 			}
384 			else
385 			{
386 				font = XComp.currentstyle.font;
387 				
388 				if ( Eco.isEmpty(font) )
389 				{
390 					font = nexacro.Component._default_font;
391 				}				
392 				
393 				text = Eco.isEmpty(text) ? XComp.text : text;
394 				multiline = Eco.isEmpty(multiline) ? (XComp["wordwrap"]? true: false) : multiline;
395 				if ( multiline && content_width == null )
396 				{
397 					content_width = XComp.getOffsetWidth();
398 				}
399 			}
400 			
401 			var size = [0, 0];
402 			if ( Eco.isEmpty(text) ) return size;  
403 
404 			if ( nexacro.Browser != "Runtime" )
405 			{
406 				var _handle = Eco._textSizeDiv;
407 				if ( !_handle )
408 				{
409 					var _doc = nexacro._managerFrameDoc;
410 					_handle = _doc.createElement("div");
411 					_doc.body.appendChild(_handle);
412 					Eco._textSizeDiv = _handle; 
413 				}
414 				var _style = _handle.style;
415 				
416 				_style.position = "absolute";
417 				_style.visibility = "hidden";	
418 				
419 				if ( multiline )
420 				{			
421 					_style.width = content_width + "px";
422 					_style.height = "auto";
423 					_style.whiteSpace = "pre-wrap";
424 					_style.wordBreak = "break-all";
425 					_style.wordWrap = "break-word";						
426 				}
427 				else
428 				{
429 					_style.width = "auto";
430 					_style.height = "auto";
431 					_style.whiteSpace = "pre";					
432 					_style.wordWrap = "normal";
433 				}
434 				
435 				_style.font = font._sysvalue;
436 				
437 				// 컴포넌트에 적용된 크기가 아닌 텍스트 자체의 크기만 구할 경우 
438 				if ( fitText == true)
439 				{
440 					_style.lineHeight = "100%";
441 				}
442 				
443 				if ( nexacro.Browser == "IE" && nexacro.BrowserVersion < 9 )
444 				{
445 					_handle.innerText = text;
446 					var rect = _handle.getBoundingClientRect();
447 					size = [rect.right - rect.left, rect.bottom - rect.top];
448 				}
449 				else
450 				{
451 					_handle.innerHTML = text;
452 					var rect = _handle.getBoundingClientRect();
453 					size = [rect.width, rect.height];
454 				}
455 				
456 				//_doc.body.appendChild(_handle);
457 				//size = [_handle.clientWidth, _handle.clientHeight];
458 				//delete handle
459 				//_doc.body.removeChild(_handle);
460 			}
461 			else
462 			{
463 				size = nexacro._getTextSize2(text, font, multiline, content_width);
464 			}
465 			
466 			size[0] = Math.ceil(size[0]);
467 			size[1] = Math.ceil(size[1]);
468 			
469 			return size;

673 						h += isNaN(itempadding.bottom) ? 0 : itempadding.bottom;

693 						}

743 		 /**
744 		 * nexacro Component 기준으로,<br>
745 		 * 지정된 size를 갖는 컴포넌트를 표시하기위한 방향 및 좌표를 반환한다.<br><br>
746 		 *  ※direction이 vert(vertical) 일때 표시방법: <br>
747 		 *    1.하단 왼쪽맞춤(default).<br>
748 		 *    2.왼쪽맞춤으로 공간부족시, 표시를 위해 팝업을 왼쪽으로 이동시킨다.<br>

752 		 *    6.영역을 초과하는 공간을 요청하면,표시가능한 x,y좌표 및 size를 반환한다.<br><br>
753 		 *  ※direction이 horz(horizontal) 일때 표시방법: <br>
754 		 *    1.우측 상단맞춤(default).<br>
755 		 *    2.상단맞춤으로 공간부족시, 표시를 위해 팝업을 위쪽으로 이동시킨다.<br>
756 		 *    3.우측에 공간이 부족하면 좌측에 표시.<br>
757 		 *    4.2번과정 반복<br>
758 		 *    5.1~4번으로도 공간확보가 안되면 direction을 무시하고 표시가능한 영역을 찾는다.<br>
759 		 *    6.영역을 초과하는 공간을 요청하면,표시가능한 x,y좌표 및 size를 반환한다.
760 		 * @param {XComp} XComp 기준이 되는 nexacro Component 
761 		 * @param {number} width 표시할 팝업 width
762 		 * @param {number} height 표시할 팝업 height
763 		 * @param {string=} direction 팝업표시 방향."vert": vertical(default),"horz": horizontal
764 		 * @param {number=} offset 표시될때 XComp와의 간격(default: 0).
765 		 * @return {array.<string...number>} [XComp 기준 팝업위치("left", "top", "right", "bottom"), x, y [ ,width, height] ]

770 		 *       [팝업위치("left", "top", "right", "bottom"), x좌표, y좌표, 가능한 width, 가능한 height]
771 		 *   </pre>
772 		 
773 		 * @example
774 		 *   var position = Eco.XComp.PositionSize.getPopupPosition(Button00, 44, 23, "vert", 2); //return: [bottom, 1394, 317]
