696 		/**

702 		 */	
703 		_logging: function(level, msg) 
704 		{
705 			var logger = Eco.Logger;
706 			var logLevel = logger.logLevel;
707 			
708 			// fatal, error 레벨은 무조건 출력
709 			if ( level > 1 && logLevel < level ) return;
710 			
711 			var message, dump, stack, elapsed, elapsedTotal, filter, showStackAnonymous;
712 			if (typeof msg != 'string')

729 
730 			var dateString = logger._getDateFormatString();
731 			var levelString = logger._LEVEL_STRING[level];
732 			var funcString = logger._getFuncFullName(callFunc);
733 			var resultMessage = dateString + " [" + levelString + "] " + funcString + " - " + message;
734 			var elapsedTime = "";
735 			if ( elapsed ) 
736 			{
737 				elapsedTime = logger._getElapsed(elapsedTotal);
738 				if ( elapsedTotal )
739 				{

741 				}
742 				else
743 				{
744 					resultMessage += " (Elapsed Time : " + elapsedTime + " sec)";
745 				}
746 			}
747 			
748 			// dump
749 			var dumpString = "";
750 			if ( dump )
751 			{
752 				dumpString = logger._getDumpString(dump);
753 				resultMessage += "\n*****************PRINTING DUMP************************";
754 				resultMessage += dumpString;
755 				resultMessage += "\n*****************COMPLETE DUMP************************";
756 			}	
757 			// callstack
758 			var stackString = "";
759 			if ( stack )
760 			{				
761 				stackString = logger._getCallStackString(logFunc, showStackAnonymous);
762 				resultMessage += "\n*****************PRINTING STACK************************";
763 				resultMessage += stackString;
764 				resultMessage += "\n*****************COMPLETE STACK************************";
765 			}
766 			
767 			// write message
768 			if ( filter )
769 			{
770 				if ( logger._filter == filter )
771 				{
772 					logger._writeLog(resultMessage);
773 				}
774 			}
775 			else
776 			{
777 				logger._writeLog(resultMessage);
778 			}
779 
780 			var buffer = logger._logBuffer;
781 			var max = logger._logBufferMax;
782 			
783 			// 지정된 로그 레벨보다 클 경우에만 buffer에 적재
784 			if ( logLevel > -1 && logLevel >= level )
785 			{
786 				if (buffer.length >= max)
787 				{
788 					buffer.splice(0, 1);
789 				}		
790 				var buff = {
791 					date: dateString,
792 					level: levelString,
793 					message: message,
794 					from:funcString,
795 					elapsed: elapsedTime,
796 					dump: dumpString,
797 					stack: stackString
798 				};
799 				logger._logBuffer.push(buff);
800 			}
801 			
802 			if ( level == 0 || level == 1 )
803 			{
804 				var e = new Error(message);
805 				throw e;				
806 			}
807 		},

820 		/**
821 		 * call stack 문자열 전체를 얻어온다.
822 		 * @private
823 		 * @param {function} func log function
824 		 * @param {boolean} showStackAnonymous stack에 익명함수(시스템 함수 포함) 보이기 여부
825 		 * @return {string} call stack 문자열 값.
826 		 * @memberOf Eco.Logger
827 		 */			
828 		_getCallStackString: function(func, showStackAnonymous)
829 		{
830 			var logger = Eco.Logger;
831 			var f = func,
832 				indstr = "",
833 				indentStr = logger._indent,
834 				ret = "",
835 				index,
836 				stackLine = "";
837 
838 			/*
839 				event handler 내에서 다른 event handler가 호출될 수 있는 구조라면
840 				무한루프가 발생된다. 따라서 stack 배열에 이미 실행한 caller
841 				정보를 담아두고 실행하기 전에 체크하여 이미 존재하는 caller 라면
842 				중단한다.
843 			*/
844 			var stack = [f];
845 			var startFunc = f.caller;
846 			
847 			while ( f )
848 			{
849 				f = f.caller;
850 				
851 				index = -1;
852 				for (var i=0,len=stack.length; i<len; i++)
853 				{
854 					if ( f == stack[i] )
855 					{
856 						index = i;
857 						break;
858 					}
859 				}
860 				
861 				if ( index < 0 )
862 				{
863 					stack.push(f);
864 					stackLine = logger._getCallStackLine(indstr, f);
865 					
866 					if ( stackLine.indexOf("anonymous function") > -1 )
867 					{
868 						if ( showStackAnonymous )
869 						{
870 							ret += "\n" + stackLine;
871 							indstr += indentStr;
872 						}
873 					}
874 					else if ( stackLine.length > 0 )
875 					{
876 						ret += "\n" + stackLine;
877 						indstr += indentStr;				
878 					}
879 				}
880 				else
881 				{
882 					ret += "\n" + indstr + "stack overflow...";
883 					break;
884 				}
885 			}
886 			
887 			return ret;
888 		},
