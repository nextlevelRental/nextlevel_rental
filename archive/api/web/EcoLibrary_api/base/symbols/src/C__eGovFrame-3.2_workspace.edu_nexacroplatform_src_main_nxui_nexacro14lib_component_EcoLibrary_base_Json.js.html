 79 		/**
 80 		* Attempt to parse an array.
 81 		* @private
 82 		*/
 83 		_parseArray : function()
 84 		{
 85 			var a = [];
 86 			var token;
 87 			var pThis = Eco.Json;
 88 			var strict = pThis._strict;
 89 			
 90 			token = pThis._nextToken();
 91 			
 92 			var tokenType = pThis._Token_Type;
 93 
 94 			if ( token.type == tokenType.RIGHT_BRACKET )
 95 			{
 96 				// we're done reading the array, so return it
 97 				return a;
 98 			}
 99 			// in non-strict mode an empty array is also a comma
100 			// followed by a right bracket
101 			else if ( !strict && token.type == tokenType.COMMA )
102 			{
103 				// move past the comma
104 				token = pThis._nextToken();
105 				// check to see if we're reached the end of the array
106 				if ( token.type == tokenType.RIGHT_BRACKET )
107 				{
108 					return a;       
109 				}
110 				else
111 				{
112 					pThis._parseError( "Leading commas are not supported.  Expecting ']' but found " + token.value );
113 				}
114 			}
115 
116 			// deal with elements of the array, and use an "infinite"
117 			// loop because we could have any amount of elements
118 			while ( true )
119 			{
120 				// read in the value and add it to the array
121 				a.push( pThis._parseValue() );
122 
123 				// after the value there should be a ] or a ,
124 				token = pThis._nextToken();
125 				if ( token.type == tokenType.RIGHT_BRACKET )
126 				{
127 					// we're done reading the array, so return it
128 					return a;
129 				}
130 				else if ( token.type == tokenType.COMMA )
131 				{
132 					// move past the comma and read another value
133 					token = pThis._nextToken();
134 
135 					// Allow arrays to have a comma after the last element
136 					// if the decoder is not in strict mode
137 					if ( !strict )
138 					{
139 						// Reached ",]" as the end of the array, so return it
140 						if ( token.type == tokenType.RIGHT_BRACKET )
141 						{
142 							return a;
143 						}
144 					}
145 				}
146 				else
147 				{
148 					pThis._parseError( "Expecting ] or , but found " + token.value );
149 				}
150 			}
151 			return null;
152 		},

155 		* Attempt to parse an object.

158 		_parseObject : function()
159 		{
160 			var pThis = Eco.Json;
161 			
162 			// create the object internally that we're going to
163 			// attempt to parse from the tokenizer
164 			var o = {};
165 			var strict = pThis._strict;
166 			// store the string part of an object member so
167 			// that we can assign it a value in the object
168 			var key;
169 
170 			// grab the next token from the tokenizer
171 			var token = pThis._nextToken();
172 			
173 			var tokenType = pThis._Token_Type;
174 
175 			// check to see if we have an empty object
176 			if ( token.type == tokenType.RIGHT_BRACE )
177 			{
178 				// we're done reading the object, so return it
179 				return o;
180 			}
181 			// in non-strict mode an empty object is also a comma
182 			// followed by a right bracket
183 			else if ( !strict && token.type == tokenType.COMMA )
184 			{
185 				// move past the comma
186 				token = pThis._nextToken();
187 				// check to see if we're reached the end of the object
188 				if ( token.type == tokenType.RIGHT_BRACE )
189 				{
190 					return o;
191 				}
192 				else
193 				{
194 					pThis._parseError( "Leading commas are not supported.  Expecting '}' but found " + token.value );
195 				}
196 			}
197 
198 			// deal with members of the object, and use an "infinite"
199 			// loop because we could have any amount of members
200 			while ( true )
201 			{
202 				if ( token.type == tokenType.STRING )
203 				{
204 					// the string value we read is the key for the object
205 					key = String( token.value );
206 					// move past the string to see what's next
207 					token = pThis._nextToken();
208 					// after the string there should be a :
209 					if ( token.type == tokenType.COLON )
210 					{
211 						// move past the : and read/assign a value for the key
212 						token = pThis._nextToken();
213 						o[key] = pThis._parseValue();  
214 						// move past the value to see what's next
215 						token = pThis._nextToken();
216 						// after the value there's either a } or a ,
217 						if ( token.type == tokenType.RIGHT_BRACE )
218 						{
219 							// we're done reading the object, so return it
220 							return o;
221 						}
222 						else if ( token.type == tokenType.COMMA )
223 						{
224 							// skip past the comma and read another member
225 							token = pThis._nextToken();
226 							// Allow objects to have a comma after the last member
227 							// if the decoder is not in strict mode
228 							if ( !strict )
229 							{
230 								// Reached ",}" as the end of the object, so return it
231 								if ( token.type == tokenType.RIGHT_BRACE )
232 								{
233 									return o;
234 								}
235 							}
236 						}
237 						else
238 						{
239 							pThis._parseError( "Expecting } or , but found " + token.value );
240 						}
241 					}
242 					else
243 					{
244 						pThis._parseError( "Expecting : but found " + token.value );
245 					}
246 				}
247 				else
248 				{
249 					pThis._parseError( "Expecting string but found " + token.value );
250 				}
251 			}
252 			return null;
253 		},

255 		_parseValue : function()
256 		{
257 			var pThis = Eco.Json;
258 			var strict = pThis._strict;
259 			var token = pThis._token;
260 			// Catch errors when the input stream ends abruptly
261 			if ( token == null )
262 			{
263 				pThis._parseError( "Unexpected end of input" );
264 			}
265 			
266 			var tokenType = pThis._Token_Type;
267 
268 			switch ( token.type )
269 			{
270 				case tokenType.LEFT_BRACE:
271 					return pThis._parseObject();
272 
273 				case tokenType.LEFT_BRACKET:
274 					return pThis._parseArray();
275 
276 				case tokenType.STRING:
277 				case tokenType.NUMBER:
278 				case tokenType.TRUE:
279 				case tokenType.FALSE:
280 				case tokenType.NULL:
281 					return token.value;
282 
283 				case tokenType.NAN:
284 					if ( !strict )
285 					{
286 						return token.value;
287 					}
288 					else
289 					{
290 						pThis._parseError( "Unexpected " + token.value );
291 					}
292 
293 				default:
294 					pThis._parseError( "Unexpected " + token.value );
295 			}
296 
297 			return null;

316 		{
317 			var pThis = Eco.Json;
318 			
319 			// determine what value is and convert it based on it's type
320 			if ( Eco.isString(value) )
321 			{
322 				// escape the string so it's formatted correctly
323 				return pThis._escapeString( value );
324 			}
325 			else if ( Eco.isNumber(value) )
326 			{
327 				// only encode numbers that finate
328 				return isFinite( value ) ? value.toString() : "null";
329 			}
330 			else if ( Eco.isBoolean(value) )
331 			{
332 				// convert boolean to string easily
333 				return value ? "true" : "false";
334 			}
335 			else if ( Eco.isArray(value) )
336 			{
337 				// call the helper method to convert an array
338 				return pThis._arrayToString( value );
339 
340 			}
341 			else if ( Eco.isXComponent(value) )
342 			{
343 				return value.name;
344 			}
345 			//else if ( value && MiscUtil.isObject(value) )
346 			else if ( value )
347 			{
348 				// call the helper method to convert an object
349 				return pThis._objectToString( value );
350 			}
351 			return "null";

354 		_escapeString : function( str )
355 		{
356 			// create a string to store the string's jsonstring value
357 			var s = "";
358 			// current character in the string we're processing
359 			var ch;
360 			// store the length in a local variable to reduce lookups
361 			var len = str.length;
362 
363 			// loop over all of the characters in the string
364 			for ( var i = 0; i < len; i++ )
365 			{
366 				// examine the character to determine if we have to escape it
367 				ch = str.charAt( i );
368 				switch ( ch )
369 				{
370 					case '"':       // quotation mark
371 						s += "\\\"";
372 						break;
373 					//case '/':     // solidus
374 					//      s += "\\/";
375 					//      break;
376 					case '\\':      // reverse solidus
377 						s += "\\\\";
378 						break;
379 					case '\b':      // bell
380 						s += "\\b";
381 						break;
382 					case '\f':      // form feed
383 						s += "\\f";
384 						break;
385 					case '\n':      // newline
386 						s += "\\n";
387 						break;
388 					case '\r':      // carriage return
389 						s += "\\r";
390 						break;
391 					case '\t':      // horizontal tab
392 						s += "\\t";
393 						break;
394 					default:        // everything else
395 						// check for a control character and escape as unicode
396 						if ( ch < ' ' )
397 						{
398 							// get the hex digit(s) of the character (either 1 or 2 digits)
399 							var hexCode = ch.charCodeAt( 0 ).toString( 16 );
400 							// ensure that there are 4 digits by adjusting
401 							// the # of zeros accordingly.
402 							var zeroPad = hexCode.length == 2 ? "00" : "000";
403 							// create the unicode escape sequence with 4 hex digits
404 							s += "\\u" + zeroPad + hexCode;
405 						}
406 						else
407 						{
408 							// no need to do any special encoding, just pass-through
409 							s += ch;
410 						}
411 				}// end switch
412 
413 			}// end for loop
414 			return "\"" + s + "\"";
415 		},

417 		_arrayToString : function( a )
418 		{
419 			var pThis = Eco.Json;
420 			// create a string to store the array's jsonstring value
421 			var s = "";
422 
423 			// loop over the elements in the array and add their converted
424 			// values to the string
425 			for ( var i = 0; i < a.length; i++ )
426 			{
427 				// when the length is 0 we're adding the first element so
428 				// no comma is necessary
429 				if ( s.length > 0 )
430 				{
431 					// we've already added an element, so add the comma separator
432 					s += ","
433 				}
434 				
435 				// convert the value to a string
436 				s += pThis._convertToString( a[i] );   
437 			}
438 
439 			// KNOWN ISSUE:  In ActionScript, Arrays can also be associative
440 			// objects and you can put anything in them, ie:
441 			//              myArray["foo"] = "bar";
442 			//
443 			// These properties aren't picked up in the for loop above because
444 			// the properties don't correspond to indexes.  However, we're
445 			// sort of out luck because the JSON specification doesn't allow
446 			// these types of array properties.
447 			//
448 			// So, if the array was also used as an associative object, there
449 			// may be some values in the array that don't get properly encoded.
450 			//
451 			// A possible solution is to instead encode the Array as an Object
452 			// but then it won't get decoded correctly (and won't be an
453 			// Array instance)
454 				
455 			// close the array and return it's string value
456 			return "[" + s + "]";

583 		getNextToken : function()
584 		{
585 			var pThis = Eco.Json;
586 			// skip any whitespace / comments since the last 
587 			// token was read
588 
589 			pThis.skipIgnored();
590 
591 			var tokenType = pThis._Token_Type;
592 
593 			var ch = pThis._ch;
594 			// examine the new character and see what we have...
595 			switch ( ch )
596 			{       
597 				case '{':
598 					pThis.setToken(tokenType.LEFT_BRACE, '{');
599 					pThis._nextChar();
600 					break;
601 				case '}':
602 					pThis.setToken(tokenType.RIGHT_BRACE, '}');
603 					pThis._nextChar();
604 					break;
605 				case '[':
606 					pThis.setToken(tokenType.LEFT_BRACKET, '[');
607 					pThis._nextChar();
608 					break;
609 				case ']':
610 					pThis.setToken(tokenType.RIGHT_BRACKET, ']');
611 					pThis._nextChar();
612 					break;
613 				case ',':
614 					pThis.setToken(tokenType.COMMA, ',');
615 					pThis._nextChar();
616 					break;
617 				case ':':
618 					pThis.setToken(tokenType.COLON, ':');
619 					pThis._nextChar();
620 					break;
621 				case 't': // attempt to read true
622 					var possibleTrue = "t" + pThis._nextChar() + pThis._nextChar() + pThis._nextChar();
623 					if ( possibleTrue == "true" )
624 					{
625 						pThis.setToken(tokenType.TRUE, true);						
626 						pThis._nextChar();
627 					}
628 					else
629 					{
630 						pThis._parseError( "Expecting 'true' but found " + possibleTrue );
631 					}
632 					break;
633 				case 'f': // attempt to read false
634 					var possibleFalse = "f" + pThis._nextChar() + pThis._nextChar() + pThis._nextChar() + pThis._nextChar();
635 					if ( possibleFalse == "false" )
636 					{
637 						pThis.setToken(tokenType.FALSE, false);
638 						pThis._nextChar();
639 					}
640 					else
641 					{
642 						pThis._parseError( "Expecting 'false' but found " + possibleFalse );
643 					}
644 					break;
645 				case 'n': // attempt to read null
646 					var possibleNull = "n" + pThis._nextChar() + pThis._nextChar() + pThis._nextChar();
647 					if ( possibleNull == "null" )
648 					{
649 						pThis.setToken(tokenType.NULL, null);
650 						pThis._nextChar();
651 					}
652 					else
653 					{
654 						pThis._parseError( "Expecting 'null' but found " + possibleNull );
655 					}
656 					break;
657 				case 'N': // attempt to read NaN
658 					var possibleNaN = "N" + pThis._nextChar() + pThis._nextChar();
659 					if ( possibleNaN == "NaN" )
660 					{
661 						pThis.setToken(tokenType.NAN, NaN);
662 						pThis._nextChar();
663 					}
664 					else
665 					{
666 						pThis._parseError( "Expecting 'NaN' but found " + possibleNaN );
667 					}
668 					break;
669 				case '"': // the start of a string
670 					pThis._readString();
671 					break;
672 				default: 
673 					// see if we can read a number
674 					if ( pThis._isDigit( ch ) || ch == '-' )
675 					{
676 						pThis._readNumber();
677 					}
678 					else if ( ch == '' )
679 					{
680 					// check for reading past the end of the string
681 						return null;
682 					}
683 					else
684 					{
685 						// not sure what was in the input string - it's not
686 						// anything we expected
687 						pThis._parseError( "Unexpected " + ch + " encountered" );
688 					}
689 			}
690 
691 			return pThis._token;
692 		},
693 
694 		_readString : function()
695 		{
696 			var pThis = Eco.Json;
697 			// the string to store the string we'll try to read
698 			var string = "";
699 
700 			// advance past the first "
701 			var ch = pThis._nextChar();
702 
703 			while ( ch != '"' && ch != '' )
704 			{
705 				// unescape the escape sequences in the string
706 				if ( ch == '\\' )
707 				{
708 					// get the next character so we know what
709 					// to unescape
710 					ch = pThis._nextChar();
711 
712 					switch ( ch )
713 					{
714 						case '"': // quotation mark
715 							string += '"';
716 							break;
717 						case '/':       // solidus
718 							string += "/";
719 							break;
720 						case '\\':      // reverse solidus
721 							string += '\\';
722 							break;
723 						case 'b':       // bell
724 							string += '\b';
725 							break;
726 						case 'f':       // form feed
727 							string += '\f';
728 							break;
729 						case 'n':       // newline
730 							string += '\n';
731 							break;
732 						case 'r':       // carriage return
733 							string += '\r';
734 							break;
735 						case 't':       // horizontal tab
736 							string += '\t'
737 							break;
738 						case 'u':
739 							// convert a unicode escape sequence
740 							// to it's character value - expecting
741 							// 4 hex digits
742 			
743 							// save the characters as a string we'll convert to an int
744 							var hexValue = "";
745 							// try to find 4 hex characters
746 							for ( var i = 0; i < 4; i++ )
747 							{
748 								// get the next character and determine
749 								// if it's a valid hex digit or not
750 								if ( !pThis._isHexDigit( ch = pThis._nextChar() ) )
751 								{
752 										pThis._parseError( " Excepted a hex digit, but found: " + ch );
753 								}
754 								// valid, add it to the value
755 								hexValue += ch;
756 							}
757 			
758 							// convert hexValue to an integer, and use that
759 							// integrer value to create a character to add
760 							// to our string.
761 							string += String.fromCharCode( parseInt( hexValue, 16 ) );
762 							break;
763 						default:
764 							// couldn't unescape the sequence, so just
765 							// pass it through
766 							string += '\\' + ch;
767 
768 					}
769 				}
770 				else
771 				{
772 					// didn't have to unescape, so add the character to the string
773 					string += ch;
774 				}
775 				// move to the next character
776 				ch = pThis._nextChar();
777 			}
778 
779 			// we read past the end of the string without closing it, which
780 			// is a parse error
781 			if ( ch == '' )
782 			{
783 				pThis._parseError( "Unterminated string literal" );
784 			}
785 
786 			// move past the closing " in the input string
787 			pThis._nextChar();
788 			pThis.setToken(pThis._Token_Type.STRING, string);
789 		},
790 
791 		_readNumber : function()
792 		{
793 			var pThis = Eco.Json;
794 			// the string to accumulate the number characters
795 			// into that we'll convert to a number at the end
796 			var input = "";
797 			var ch = pThis._ch;
798 			var strict = pThis._strict;
799 			// check for a negative number
800 			if ( ch == '-' )
801 			{
802 				input += '-';
803 				ch = pThis._nextChar();
804 			}
805 
806 			// the number must start with a digit
807 			if ( !pThis._isDigit( ch ) )
808 			{
809 				pThis._parseError( "Expecting a digit" );
810 			}
811 
812 			// 0 can only be the first digit if it
813 			// is followed by a decimal point
814 			if ( ch == '0' )
815 			{
816 				input += ch;
817 				ch = pThis._nextChar();
818 
819 				// make sure no other digits come after 0
820 				if ( pThis._isDigit( ch ) )
821 				{
822 					pThis._parseError( "A digit cannot immediately follow 0" );
823 				}
824 				// unless we have 0x which starts a hex number, but this
825 				// doesn't match JSON spec so check for not strict mode.
826 				else if ( !strict && ch == 'x' )
827 				{
828 					// include the x in the input
829 					input += ch;
830 					ch = pThis._nextChar();
831 
832 					// need at least one hex digit after 0x to
833 					// be valid
834 					if ( pThis._isHexDigit( ch ) )
835 					{
836 						input += ch;
837 						ch = pThis._nextChar();
838 					}
839 					else
840 					{
841 						pThis._parseError( "Number in hex format require at least one hex digit after \"0x\"" );       
842 					}
843 
844 					// consume all of the hex values
845 					while ( pThis._isHexDigit( ch ) )
846 					{
847 						input += ch;
848 						ch = pThis._nextChar();
849 					}
850 				}
851 			}
852 			else
853 			{
854 				// read numbers while we can
855 				while ( pThis._isDigit( ch ) )
856 				{
857 					input += ch;
858 					ch = pThis._nextChar();
859 				}
860 			}
861 			// check for a decimal value
862 			if ( ch == '.' )
863 			{
864 				input += '.';
865 				ch = pThis._nextChar();
866 				// after the decimal there has to be a digit
867 				if ( !pThis._isDigit( ch ) )
868 				{
869 					pThis._parseError( "Expecting a digit" );
870 				}
871 				// read more numbers to get the decimal value
872 				while ( pThis._isDigit( ch ) )
873 				{
874 					input += ch;
875 					ch = pThis._nextChar();
876 				}
877 			}
878 
879 			// check for scientific notation
880 			if ( ch == 'e' || ch == 'E' )
881 			{
882 				input += "e"
883 				ch = pThis._nextChar();
884 				// check for sign
885 				if ( ch == '+' || ch == '-' )
886 				{
887 					input += ch;
888 					ch = pThis._nextChar();
889 				}
890 
891 				// require at least one number for the exponent
892 				// in this case
893 				if ( !pThis._isDigit( ch ) )
894 				{
895 					pThis._parseError( "Scientific notation number needs exponent value" );
896 				}
897 				// read in the exponent
898 				while ( pThis._isDigit( ch ) )
899 				{
900 					input += ch;
901 					ch = pThis._nextChar();
902 				}
903 			}
904 
905 			// convert the string to a number value
906 			var num = Number( input );
907 
908 			if ( isFinite( num ) && !isNaN( num ) )
909 			{
910 				// the token for the number that we've read
911 				pThis.setToken(pThis._Token_Type.NUMBER, num);
912 				return;
913 			}
914 			else
915 			{
916 				pThis._parseError( "Number " + num + " is not valid!" );
917 			}
918 			
919 			pThis._token = null;
920 		},
921 
922 		_nextChar : function()
923 		{
924 			var pThis = Eco.Json;
925 			return pThis._ch = pThis._jsonString.charAt( pThis._loc++ );
926 		},

928 		skipIgnored : function()
929 		{
930 			var originalLoc;
931 			var pThis = Eco.Json;
932 			// keep trying to skip whitespace and comments as long
933 			// as we keep advancing past the original location 
934 			do
935 			{
936 				originalLoc = pThis._loc;
937 				pThis._skipWhite();
938 				pThis._skipComments();
939 			}
940 			while ( originalLoc != pThis._loc );
941 		},

943 		_skipComments : function()
944 		{
945 			var pThis = Eco.Json;
946 			var ch = pThis._ch;
947 			if ( ch == '/' )
948 			{
949 				// Advance past the first / to find out what type of comment
950 				ch = pThis._nextChar();
951 				switch ( ch )
952 				{
953 					case '/': // single-line comment, read through end of line
954 
955 						// Loop over the characters until we find
956 						// a newline or until there's no more characters left
957 						do
958 						{
959 							ch = pThis._nextChar();
960 						}
961 						while ( ch != '\n' && ch != '' )
962 
963 						// move past the \n
964 						ch = pThis._nextChar();
965 						break;
966 
967 					case '*': // multi-line comment, read until closing */
968 						// move past the opening *
969 						ch = pThis._nextChar();
970 
971 						// try to find a trailing */
972 						while ( true )
973 						{
974 							if ( ch == '*' )
975 							{
976 								// check to see if we have a closing /
977 								ch = pThis._nextChar();
978 								if ( ch == '/')
979 								{
980 									// move past the end of the closing */
981 									ch = pThis._nextChar();
982 									break;
983 								}
984 							}
985 							else
986 							{
987 								// move along, looking if the next character is a *
988 								ch = pThis._nextChar();
989 							}
990 							// when we're here we've read past the end of 
991 							// the string without finding a closing */, so error
992 							if ( ch == '' )
993 							{
994 								pThis._parseError( "Multi-line comment not closed" );
995 							}
996 						}
997 						break;
998 					// Can't match a comment after a /, so it's a parsing error
999 					default:
1000 						pThis._parseError( "Unexpected " + ch + " encountered (expecting '/' or '*' )" );
1001 				}
1002 			}
1003 		},
1004 
1005 		_skipWhite : function()
1006 		{
1007 			// As long as there are spaces in the input 
1008 			// stream, advance the current location pointer
1009 			// past them
1010 			
1011 			var pThis = Eco.Json;
1012 			var ch = pThis._ch;
1013 			while ( pThis._isWhiteSpace( ch ) )
1014 			{
1015 				ch = pThis._nextChar();
1016 			}
1017 		},

1019 		_isWhiteSpace : function( ch )
1020 		{
1021 			return ( ch == ' ' || ch == '\t' || ch == '\n' || ch == '\r' );
1022 		},
1023 
1024 		_isDigit : function( ch )
1025 		{
1026 			return ( ch >= '0' && ch <= '9' );
1027 		},
1028 
1029 		_isHexDigit : function( ch )
1030 		{
1031 			// get the uppercase value of ch so we only have
1032 			// to compare the value between 'A' and 'F'
1033 			var uc = ch.toUpperCase();
1034 
1035 			// a hex digit is a digit of A-F, inclusive ( using
1036 			// our uppercase constraint )
1037 			return ( Eco.Json._isDigit( ch ) || ( uc >= 'A' && uc <= 'F' ) );
1038 		},
