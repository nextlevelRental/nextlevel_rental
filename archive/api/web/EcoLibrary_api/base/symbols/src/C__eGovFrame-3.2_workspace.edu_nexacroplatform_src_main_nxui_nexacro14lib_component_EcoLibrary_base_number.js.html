102 		* 실수인 경우에는 기억공간의 범위를 벗어나는 결과가 나타나면 나머지수를 버립니다.<br>

104 		* 예를들어,<br>

519 		*     precision = 2,
520 		*     unit = false;
521 		* var result = Eco.number.bytesToSize(size, precision, unit);
522 		* trace(result); // output : 976.00
523 		* result = Eco.number.bytesToSize(size, 0);
524 		* trace(result); // output : 976 MB
525 		* result = Eco.number.bytesToSize(size);
526 		* trace(result); // output : 976.0 MB
527 		* @memberOf Eco.number
528 		*/	
529 		bytesToSize: function(bytes, precision, unit) 
530 		{
531 			if(precision === null || precision === undefined || precision === "") precision = 1;
532 
533 			if(unit !== false) unit = true;
534 			
535 			var units = ['B', 'KB', 'MB', 'GB', 'TB', 'PB', 'EB', 'ZB', 'YB'];
536 			var i = 0;
537 			while(bytes >= 1024) 
538 			{
539 				bytes /= 1024;
540 				++i;
541 			}
542 			
543 			if(!unit)
544 			{
545 				return bytes.toFixed(precision);
546 			} 
547 			else 
548 			{
549 				if ( precision > 0 )
550 				{
551 					return bytes.toFixed(precision) + ' ' + units[i];
552 				}
553 				else
554 				{
555 					return Math.round(bytes) + ' ' + units[i];
556 				}
557 			}

561 		/**
562 		* 반올림할 자리의 값이 1~4인 경우는 버리고, 6~9인 경우는 올림 처리한다.<br>
563 		* 5의 경우 그 앞자리 값이 짝수이면 버리고, 홀수이면 올림 처리한다.<br>
564 		* 편중 오차를 최소화하는 방법 중의 하나로 Banker's Rounding을 사용하는 것입니다.
565 		* @param {number} value 변환할 값.
566 		* @param {number} length 소수부.
567 		* @return {number} 
568 		* @example
569 		* alert(Eco.number.evenRound(1.5)); // 2
570 		* alert(Eco.number.evenRound(2.5)); // 2
571 		* alert(Eco.number.evenRound(1.535, 2)); // 1.54
572 		* alert(Eco.number.evenRound(1.525, 2)); // 1.52
573 		* @memberOf Eco.number
574 		*/		
575 		evenRound: function(num, decimalPlaces)
576 		{
577 			var d = decimalPlaces || 0;
578 			var m = Math.pow(10, d);
579 			var n = +(d ? num * m : num).toFixed(8); // Avoid rounding errors
580 			var i = Math.floor(n), f = n - i;
581 			var r = (f == 0.5) ? ((i % 2 == 0) ? i : i + 1) : Math.round(n);
582 			return d ? r / m : r;
583 		},

593 		 * mask format 변환 정보 반환.
594 		 * @private

596 		 * @return {array} mask format 변환 정보
597 		 * @memberOf Eco.number
598 		 */	
599 		_parseNumberMask: function(strMask)
600 		{
601 			var pThis = Eco.number;
602 			
603 			var res = pThis._numberMaskCache[strMask];
604 			if ( res ) return res;
605 		
606 			var dispComma = false,
607 				nMin = 0,
608 				nMax = 0,
609 				nDecimalMin = -1,
610 				nDecimalMax = 0;
611 
612 			strMask = nexacro.trimLeft(strMask);
613 
614 			if (strMask.length <= 0 ) 
615 			{
616 				return [dispComma, nMin, nMax, nDecimalMin, nDecimalMax];
617 			}
618 
619 			nDecimalMax = 0;
620 			nDecimalMin = 0;
621 			dispComma = (strMask.indexOf(",") >= 0);
622 
623 			var c,
624 				bFindPoint = false;
625 
626 			for( var i = 0, n = strMask.length; i < n; i++ ) 
627 			{
628 				c = strMask.charAt(i);
629 				if ( c == '.' )
630 				{
631 					bFindPoint = true;
632 				}
633 				else if ( c == '#' || c == '0' || c == '9' || c == ',' ) 
634 				{
635 					if ( bFindPoint ) 
636 					{
637 						if ( c == '0' ) 
638 						{
639 							nDecimalMax++;
640 							nDecimalMin = nDecimalMax;
641 						}
642 						else
643 						{
644 							nDecimalMax++;
645 						}
646 					}
647 					else 
648 					{
649 						if ( c == '0' )
650 						{
651 							nMin++;
652 						}
653 						else if ( nMin > 0)
654 						{
655 							nMin++;
656 						}
657 						nMax++;
658 					}
659 				}
660 			}// end of for.
661 			
662 			res = [dispComma, nMin, nMax, nDecimalMin, nDecimalMax];
663 			pThis._numberMaskCache[strMask] = res;
664 			return res;
665 		},

713 		 * mask 제거.
714 		 * @private
715 		 * @param {string} strText 숫자형 문자.
716 		 * @return {string} 변환된 문자열
717 		 * @memberOf Eco.number
718 		 */	
719 		_removeMask: function(str) 
720 		{
721 			var pThis = Eco.number;
722 			
723 			str = str.trim();
724 			var ntxtLen = str.length,
725 				i,
726 				pointVal = pThis.point,
727 				separatorVal = pThis.separator;
728 				bPoint = false, // 소숫점은 1개만 인정.
729 				bInside = false, // 부호는 숫자가 나오기 전에만 인정.
730 				buf = [];
731 			for(i = 0 ; i < ntxtLen ; i++ ) 
732 			{
733 				var c = str.charAt(i);
734 				if( ( c == '+' || c == '-') && ( bInside === false) ) 
735 				{
736 					// 부호는 숫자가 나오기 전에만 인정.
737 					buf.push(c);
738 					bInside = true;
739 				}
740 				else if( nexacro.isNumeric(c) ) 
741 				{
742 					// 숫자인경우 인정.
743 					buf.push(c);
744 					bInside = true;
745 				}
746 				else if( c == pointVal && bPoint === false ) 
747 				{
748 					// 소숫점은 1회만 인정.
749 					buf.push(c);
750 					bPoint = true;
751 					bInside = true;
752 				}
753 				else if( c != separatorVal ) 
754 				{
755 					return "";
756 				}
757 			}
758 			return buf.join("");
759 		},

762 		 * 숫자 단위 적용.
763 		 * @private
764 		 * @param {string} strNumber 숫자형 문자.
765 		 * @return {string} 변환된 문자열
766 		 * @memberOf Eco.number
767 		 */
768 		_applyComma: function(strNumber) 
769 		{
770 			// 차후에 locale 처리 필요.
771 			
772 			var pThis = Eco.number;
773 			
774 			var grouping = pThis.separatorGrouping;
775 			var thousands_sep = pThis.separator;
776 			if( thousands_sep.length > 0 )
777 			{
778 				var	dec_buf = strNumber.split(""),
779 					dec_size = strNumber.length,
780 					out_size = (thousands_sep.length + 1) * strNumber.length,
781 					out_buf = [],
782 					cur_group = 0, d_size = dec_size,
783 					endpos = out_size,
784 					groupingIdx = 0,
785 					g;
786 
787 				grouping = grouping.split("");
788 
789 				while (grouping[groupingIdx] && d_size > 0 ) 
790 				{	
791 					g = grouping[groupingIdx];
792 					if ( g == "\\" ) 
793 					{
794 						groupingIdx++;
795 						g = parseInt(grouping[groupingIdx]);
796 					}
797 					if (g > 0 ) 
798 					{
799 						cur_group = g;
800 						while (g-- > 0 && d_size > 0)
801 							out_buf[--endpos] = dec_buf[--d_size];
802 						if (d_size > 0)
803 							out_buf[--endpos] = thousands_sep;
804 					}
805 					else if (g == 0 && d_size > cur_group) 
806 					{
807 						g = cur_group;
808 						while (g-- > 0)
809 							out_buf[--endpos] = dec_buf[--d_size];
810 						if (d_size > 0)
811 							out_buf[--endpos] = thousands_sep;
812 					}
813 					else if (g == 0 && d_size <= cur_group && d_size > 0 ) 
814 					{
815 						g = d_size;
816 						while (g-- > 0)
817 							out_buf[--endpos] = dec_buf[--d_size];
818 					}
819 					else 
820 					{
821 						break;
822 					}
823 				}
824 				return out_buf.slice(endpos,out_size + endpos).join("");
825 			}
826 			return strNumber;
827 		},

829 		/**
830 		 * 숫자형 변환
831 		 * @private
832 		 * @param {string} strValue 숫자형 문자.
833 		 * @param {boolean} bTrim 공백제거 여부.
834 		 * @return {string} 변환된 문자열.
835 		 * @memberOf Eco.number
836 		 */	
837 		_normalizeValue: function(strValue, bTrim) 
838 		{
839 			var pThis = Eco.number;
840 			
841 			if (strValue.length<=0) return strValue;
842 			if ( bTrim === undefined ) bTrim = true;
843 				
844 			// 좌우 공백 제거, 마스크 제거.
845 			strValue = pThis._removeMask(strValue);
846 			// 정규형식으로 만듬.
847 			if( strValue.charAt( strValue.length -1 ) == pThis.point )
848 				strValue = strValue + "0";
849 
850 			if(bTrim) 
851 			{
852 				// 0 Trim 하기 전에 부호 있는 경우 주의.
853 				var bSign = ( strValue.charAt(0) == '-' ) ? 1 : 0;
854 				var pointVal = pThis.point;
855 
856 				// 0000~~ 에 대해서 첫 자리 0을 삭제
857 				// -0000~~ 에 대해서 부호 뒷 자리 0을 삭제.
858 				while( strValue.charAt( 0+bSign ) == '0' 
859 						&& strValue.charAt( 1+bSign ) != pointVal
860 						&& strValue.length != (1+bSign) ) 
861 				{
862 						strValue = strValue.substring(0, 0+bSign) + strValue.substr(0+bSign+1, strValue.length);
863 				}
864 				// .~~0000에 대해서 0을 Trim
865 				var nPoint = strValue.indexOf(pointVal);
866 				if( nPoint >= 0 ) 
867 				{
868 					var i;
869 					for( i = strValue.length - 1 ; i > nPoint+1 ; i -- ) 
870 					{
871 						if( strValue.charAt( i ) != '0' )
872 								break;
873 					}
874 					strValue = strValue.substring(0, i + 1 );
875 				}
876 			}
877 
878 			// 부호에 대해 정규형으로.
879 			if( strValue.charAt(0) == '+' ) 
880 			{
881 				strValue = strValue.substr(1);
882 			}
883 			else if( strValue.length <= 0 || parseFloat(strValue) == 0.0 ) 
884 			{
885 				// 값이 0.0인경우 부호 제거.
886 				if( strValue.charAt(0) == '-' )
887 					strValue = strValue.substr(1);
888 			}
889 			return strValue;
890 		},

894 		 * @param {string|number} value 숫자형식 값
895 		 * @param {string} strMask mask할 format 문자열.
896 		 * @return {string} 변환된 문자열.
897 		 * @example
898 		 * var val = 1234.567;
899 		 * var result = Eco.number.getMaskFormatString(val, "99.99");
900 		 * trace(result); // output : 1234.56
901 		 * result = Eco.number.getMaskFormatString(val, "9900.0099");
902 		 * trace(result); // output : 1234.567
903 		 * result = Eco.number.getMaskFormatString(val, "9,999.9");
904 		 * trace(result); // output : 1,234.5
905 		 *
906 		 * val = 1.2;
907 		 * result = Eco.number.getMaskFormatString(val, "99.99");
908 		 * trace(result); // output : 1.2
909 		 * result = Eco.number.getMaskFormatString(val, "9900.0099");
910 		 * trace(result); // output : 01.20
911 		 * result = Eco.number.getMaskFormatString(val, "9,999.9");
912 		 * trace(result); // output : 1.2
913 		 * @memberOf Eco.number
914 		 */		
915 		getMaskFormatString: function(value, strMask)
916 		{
917 			var pThis = Eco.number;
918 			
919 			var strText = value + "";
920 			strText = pThis._normalizeValue(strText, true);
921 
922 			var nDecimalLen = 0;		// 소수부 길이.
923 			var nIntegerLen = 0;		// 정수부 길이.
924 
925 			// Split : value를 정수부, 소수부로 나누고 소수점 유무, 양음부호부 까지 판별한다.
926 			var ret = pThis._numberSplit(strText),
927 				nSign = ret[0],
928 				bPoint = ret[1],
929 				strNumber = ret[2],
930 				strDecimal = ret[3];
931 
932 			// 각 부분의 길이구함.
933 			nIntegerLen = strNumber.length;
934 			nDecimalLen = strDecimal ? strDecimal.length : 0;
935 
936 			var maskInfo = pThis._parseNumberMask(strMask),
937 				dispComma = maskInfo[0],
938 				nMin = maskInfo[1],
939 				nMax = maskInfo[2],
940 				nDecimalMin = maskInfo[3],
941 				nDecimalMax = maskInfo[4];
942 
943 			if (nMin > nIntegerLen) 
944 			{
945 				var tmpStr = "";
946 				for(var i=0, n =  nMin - nIntegerLen; i < n ; i++) 
947 				{
948 					tmpStr += "0";
949 				}
950 				strNumber = tmpStr + strNumber;
951 			}
952 
953 			if (nDecimalMin > nDecimalLen) 
954 			{
955 				var tmpStr = "";
956 				for(var i=0, n = nDecimalMin - nDecimalLen; i < n ; i++) 
957 				{
958 					tmpStr += "0";
959 				}
960 				strDecimal = strDecimal + tmpStr;
961 			}
962 			else if ( nDecimalMax != -1 && nDecimalMax < nDecimalLen) 
963 			{
964 				strDecimal = strDecimal.substring(0, nDecimalMax) + strDecimal.substr(nDecimalLen, strDecimal.length);
965 			}
966 			// 정수부분에 Comma적용.
967 			if(  dispComma )
968 				strNumber = pThis._applyComma(strNumber);
969 
970 			if( nSign < 0 ) strText = "-";
971 			else if( nSign > 0) strText = "+";
972 			else	strText = "";
973 
974 			// 정수부.
975 			strText += strNumber;
976 
977 			// 소수부.
978 			if( strDecimal.length > 0 ) 
979 			{
980 				strText += pThis.point;
981 				strText += strDecimal;
982 			}
983 			return strText;
984 		}
