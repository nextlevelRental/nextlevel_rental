 39 				leftWidth   = nexacro.toNumber(leftWidth.replace("px",""));
 40 				topWidth    = nexacro.toNumber(topWidth.replace("px",""));
 41 				rightWidth  = nexacro.toNumber(rightWidth.replace("px",""));
 42 				bottomWidth = nexacro.toNumber(bottomWidth.replace("px",""));
 43 				
 44 				return [leftWidth, topWidth, rightWidth, bottomWidth];
 45 			}
 46 			else

 48 				return [0, 0, 0, 0];

108 		 /**
109 		 * nexacro Component의 Style을 반환한다.
110 		 * @param {XComp} xComp nexacro Component
111 		 * @return {object} 
112 		 * @example
113 		 * trace(Eco.XComp.Style.getStyle(xComp));
114          * // output:
115          * //  {background: "left middle"
116          * //   ,border: "1 solid #808080ff "

118          * //   ,align: "center middle"}
119          *
120          * @memberOf Eco.XComp.Style
121 		 */
122 		getStyle: function(xComp)	
123 		{
124 			var style = xComp.style;
125 			var json = {};
126 			var isFunction = Eco.isFunction;
127 			
128 			for ( var name in style )
129 			{
130 				if ( style.hasOwnProperty(name) )
131 				{
132 					if ( name.charAt(0) == "_" ) continue;
133 					
134 					var styleProp = style[name];
135 					if ( styleProp )
136 					{
137 						if ( !isFunction(styleProp) )
138 						{
139 							if(styleProp === undefined) continue;
140 							
141 							value = styleProp.toString();
142 							
143 							if(value == "undefined") continue;
144 							
145 							if ( value.length )
146 							{
147 								json[name] = value;
148 							}
149 						}
150 					}
151 				}
152 			}
153 			
154 			return json;

294 		/**
295 		* nexacro Component에 argument로 주어진 styleValue를 condition에 따라 적용한다.<br><br>
296 		* - condition이 string일 때:<br>
297 		*    styleValue의 속성과 동일한 xComp의 속성값이 condition과 같을 때 적용한다.<br>
298 		* - condition이 function일 때:<br>
299 		*    condition의 실행값이 true일 경우에만 적용한다.<br>
300 		*    ※ 이 때 condition 함수를 호출 시 xComp, 속성명, 속성값이 arguments로 전달된다.<br>
301 		* - condition이 ""(빈문자열) 또는 null, undefined 일 때:<br>
302 		*    styleValue의 속성명과 동일한 xComp의 속성값이 없을 경우에만 적용한다.<br><br>
303 		* ※component의 style에 지정된 속성이 없어서 테마의 기본값이 표시되는 것은<br>
304 		*   속성값이 없는 것으로 본다.
305 		* @param {XComp} xComp target 객체 
306 		* @param {object} styleValue source 객체
307 		* @param {function|string|undefined} condition 적용조건
308 		* @param {object} scope callback 함수에 대한 수행 scope(this) 
309 		* @param {boolean} clearFlag 모든 style값 clear여부(default:true).
310 		* @example
311 		* //btn_result_01의 color속성이 'red'일 경우에 이 속성값을 "blue"로 적용
312 		* var styleValue = {color:"blue"};
313 		* Eco.XComp.Style.setStyleIf(btn_result_01, styleValue, "red", this, false);
314         *
315         * //btn_result_02의 enable속성이 true일 경우에만 boder,color 적용
316         * var styleValue = {border:"3 double red", color: "red"};
317         * Eco.XComp.Style.setStyleIf(btn_result_02, styleValue, function(xComp, name, value) {
318         *         if (xComp.enable == true)
319         *         {
320         *             return true;
321         *         }
322         *         return false;
323         *     }
324         * , this);
325         *
326         * //btn_result_03의 background와 align이 설정되지 않은 경우 
327         * //아래의 styleValue 값으로 적용된다.
328         * var styleValue = {background:"orange", align:"right middle"};
329         * Eco.XComp.Style.setStyleIf(btn_result_03, styleValue, "" , this, false);
330         *
331 		* @memberOf Eco.XComp.Style
332 		*/
333 		setStyleIf: function(xComp, styleValue, condition, scope, clearFlag)
334 		{
335 			if(condition === null || condition === undefined) condition = "";
336 			
337 			var pThis = Eco.XComp.Style;
338 			
339 			if (xComp && styleValue) 
340 			{
341 			
342 				if(Eco.isEmpty(clearFlag)) clearFlag = true;
343 				
344 				//적용할 style 구문 추출.
345 				if(clearFlag)
346 				{
347 					//기존 style값 clear
348 					var prevStyle = pThis.getStyle(xComp);
349 					for(var name in prevStyle)
350 					{
351 						if ( prevStyle.hasOwnProperty(name) )
352 						{
353 							//xComp[name] = "";
354 							if(name.substr(0,1) == "_") continue;
355 							xComp["style"]["set_" + name]("");
356 						}
357 					}
358 				}
359 			
360 				var value, ret,
361 					isFunc = false;
362 				
363 				if (condition && Eco.isFunction(condition))
364 				{
365 					isFunc = true;
366 				}
367 				
368 				var xCompStyle = xComp.style;
369 				for (var name in styleValue)
370 				{
371 					if(name.substr(0,1) == "_") continue;
372 					
373 					if (styleValue.hasOwnProperty(name))
374 					{
375 						value = styleValue[name];
376 						
377 						if (isFunc)
378 						{
379 							if(condition.call(scope, xComp, name, value))
380 							{
381 								xComp["style"]["set_" + name](value);
382 							}
383 						}
384 						//else if (xCompStyle[name].toString() == condition.toString())
385 						else if (condition == "" && Eco.isEmpty(xCompStyle[name]))
386 						{
387 							xComp["style"]["set_" + name](value);
388 						}						
389 						else if (xCompStyle[name] == condition)
390 						{
391 							xComp["style"]["set_" + name](value);
392 						}						
393 					}
394 				}
395 			}
396 		}
