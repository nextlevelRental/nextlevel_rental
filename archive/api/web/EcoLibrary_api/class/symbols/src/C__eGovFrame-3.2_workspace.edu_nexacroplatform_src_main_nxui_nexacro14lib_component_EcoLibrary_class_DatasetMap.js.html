  9 		/**
 10 		 * DatasetMap 생성자(constructor)
 11 		 * @class DatasetMap
 12 		 * @classdesc Dataset에서 행(row)별로 데이터를 unique한 key column으로 map 정보를 구성하는 객체<br>
 13 		 * key 항목으로 검색 처리가 for loop으로 하지 않고 object key collection 검색하여 처리 속도가 데이터의 크기와 상관없다.<br>
 14 		 * 또한 Dataset의 행(row)별로 객체 값으로 데이터를 구성하여 담을 수 있는 메모리 구조이다.
 15 		 * @constructor DatasetMap
 16 		*/
 17 		initialize: function()
 18 		{
 19 			this.clear();
 20 			this._indexMapNeedRefresh = true;
 21 			return this;
 22 		},
 23 		/**
 24 		 * 데이터를 모두 지운다.
 25 		 * @memberOf DatasetMap
 26 		*/
 27 		clear: function()
 28 		{
 29 			if ( !this.datas || this.datas.length )
 30 			{
 31 				this.datas = [];
 32 				this.keyMap = {};
 33 				this._indexMap = null;
 34 				this._rebuildIndexMap = true;
 35 			}
 36 		},
 37 		/**
 38 		 * 주어진 key 값이 존재하는지 확인하는 함수
 39 		 * @param {string} key key 값
 40 		 * @return {boolean} 존재 여부
 41 		 * @memberOf DatasetMap
 42 		*/
 43 		containsKey: function(key)
 44 		{
 45 			return this.keyMap.hasOwnProperty(key+"");
 46 		},
 47 		/**
 48 		 * 주어진 key 값이 가지는 row를 얻는다.
 49 		 * @param {string} key key 값
 50 		 * @return {number} key값에 해당하는 row
 51 		 * @memberOf DatasetMap
 52 		*/
 53 		findRowByKey: function(key)
 54 		{
 55 			if ( this._rebuildIndexMap )
 56 			{
 57 				var indexMap = {},
 58 					datavals = this.datas,
 59 					data;
 60 				for (var i = 0, len = datavals.length; i < len; i++)
 61 				{
 62 					data = datavals[i];
 63 					indexMap[data.key] = i;
 64 				}
 65 				this._indexMap = indexMap;
 66 				this._rebuildIndexMap = false;
 67 			}
 68 			var row = this._indexMap[key + ""];
 69 			return row == null ? -1 : row;
 70 		},
 71 		/**
 72 		 * 주어진 row에 주어진 key, data을 insert한다.
 73 		 * @param {number} row index of row 
 74 		 * @param {string} key key 값
 75 		 * @param {*} data row별로 저장하는 객체 값
 76 		 * @param {boolean} checkLength row 유효성 검사를 할 것인지 여부(default: true)
 77 		 * @memberOf DatasetMap
 78 		*/
 79 		insert: function(row, key, data, checkLength)
 80 		{
 81 			if ( this.containsKey(key) ) return;
 82 			data.key = key + "";
 83 			if ( checkLength === false )
 84 			{
 85 				this.keyMap[key + ""] = data;
 86 				this.datas[row] = data;
 87 			}
 88 			if ( checkLength || ( row > -1 && row < this.datas.length ) )
 89 			{
 90 				this.keyMap[key + ""] = data;
 91 				this.datas.splice(row, 0, data);
 92 			}
 93 			else
 94 			{
 95 				this.add(key, data);
 96 			}
 97 			this._rebuildIndexMap = true;
 98 		},
 99 		/**
100 		 * 주어진 key, data을 add한다.
101 		 * @param {string} key key 값
102 		 * @param {*} data row별로 저장하는 객체 값
103 		 * @memberOf DatasetMap
104 		*/
105 		add: function(key, data)
106 		{
107 			key = key + "";
108 			if ( this.containsKey(key) ) return;
109 			data.key = key;
110 			this.keyMap[key] = data;
111 			this.datas.push(data);
112 			this._rebuildIndexMap = true;
113 		},
114 		/**
115 		 * 주어진 row의 data를 삭제한다.
116 		 * @param {number} row index of row 
117 		 * @memberOf DatasetMap
118 		*/
119 		removeAt: function(row)
120 		{
121 			var data = this.datas[row];
122 			if ( !data ) return false;
123 			var key = data.key;
124 			if ( !this.containsKey(key) ) return false;
125 			delete this.keyMap[key];
126 			this.datas.splice(row, 1);
127 			this._rebuildIndexMap = true;
128 		},
129 		/**
130 		 * 주어진 key의 data를 삭제한다.
131 		 * @param {string} key 값
132 		 * @memberOf DatasetMap
133 		*/
134 		removeAtKey: function(key)
135 		{
136 			key = key + "";
137 			if ( !this.containsKey(key) ) return false;
138 			var row = this.findRowByKey(key);
139 			if ( row == -1 ) return false;
140 
141 			delete this.keyMap[key];
142 			this.datas.splice(row, 1);
143 			this._rebuildIndexMap = true;
144 		},
145 		/**
146 		 * 주어진 dataset으로 내부 데이터를 구성한다.<br>
147 		 * 두번째 주어지는 keyColumnId 이 dataset에 구성되는 column명이고 이 column에 존재하는 데이터가 key 값이 된다.<br>
148 		 * 세번쨰 인자로 주어지는 createDataFunc는 data 객체를 구성하기 호출하는 함수이다.<br>
149 		 * 이 함수는 정의시에 return 값으로 객체가 되어야 한다.<br>
150 		 * createDataFunc의 arguments ( dataset, row, key ) 로 넘겨준다.
151 		 * @example
152 		 * //taskItem 객체 생성하는 함수
153 		 * function createTaskItem(ds, row, key)
154 		 * {
155 		 *    var sdt = ds.getColumn(row, "Start");
156 		 *    var edt = ds.getColumn(row, "Finish");
157 		 *    var taskName = ds.getColumn(row, "TaskName");
158 		 *    var data = {
159 		 *      taskname: taskName,
160 		 *      startdate: sdt,
161 		 *      finishdate: edt
162 		 *    };
163 		 *    return data;
164 		 * }
165 		 *
166 		 * var dsMap = new DatasetMap();
167 		 * dsMap.buildDataWithDataset(Dataset00, "cd", createTaskItem, this);
168 		 *
169 		 * @param {Dataset} ds dataset
170 		 * @param {string} keyColumnId key 값을 얻는 dataset column 명
171 		 * @param {function} createDataFunc data 객체를 생성하는 처리함수
172 		 * @param {*} scope createDataFunc 함수 내부에 사용되는 this
173 		 * @memberOf DatasetMap
174 		*/
175 		buildDataWithDataset: function(ds, keyColumnId, createDataFunc, scope)
176 		{
177 			this.clear();
178 			var key, data,
179 				datavals = this.datas,
180 				keyvals = this.keyMap;
181 			for ( var i = 0, len = ds.rowcount ; i < len ; i++ )
182 			{
183 				key = ds.getColumn(i, keyColumnId) + "";
184 				data = createDataFunc.call(scope, ds, i, key);
185 				data.key = key;
186 				datavals[i] = data;
187 				keyvals[key] = data;
188 			}
189 			this._rebuildIndexMap = true;
190 		},
191 		/**
192 		 * 주어진 행(row)의 data 객체를 구하는 메소드입니다.
193 		 * @param {number} row index of row
194 		 * @return {*} 주어진 row의 data 객체
195 		 * @memberOf DatasetMap
196 		*/
197 		getAt: function(row)
198 		{
199 			return this.datas[row];
200 		},
201 		/**
202 		 * 주어진 key의 data 객체를 구하는 메소드입니다.
203 		 * @param {string} key key 값
204 		 * @return {*} 주어진 key의 data 객체
205 		 * @memberOf DatasetMap
206 		*/
207 		getByKey: function(key)
208 		{
209 			return this.keyMap[key];
210 		},
211 		/**
212 		 * 주어진 oldRow의 data 객체를 주어진 newRow로 이동하는 처리이다.
213 		 * @param {number} oldRow 이동 하려는 row
214 		 * @param {number} newRow 이동을 원하는 row
215 		 * @memberOf DatasetMap
216 		*/
217 		moveRow: function(oldRow, newRow)
218 		{
219 			var dataLen = this.datas.length;
220 			if ( oldRow < 0 || oldRow >= dataLen ) return;
221 			if ( newRow < 0 || newRow > dataLen ) return;
222 
223 			if ( oldRow > newRow )
224 			{
225 				var data = this.datas[oldRow];
226 				this.datas.splice(newRow, 0, data);
227 				this.datas.splice(oldRow + 1, 1);
228 				this._rebuildIndexMap = true;
229 			}
230 			else if ( oldRow < newRow )
231 			{
232 				var data = this.datas[oldRow];
233 				this.datas.splice(oldRow, 1);
234 				this.datas.splice(newRow - 1, 0, data);
235 				this._rebuildIndexMap = true;
236 			}
237 		}
238 	}); // end of 'JsNamespace.declare("DatasetMap",'
