208 		/**
209 		 * where 조건문의 token을 얻어온다.<br>
210 		 * thanks to Douglas Crockford. tokens.js (crockford.com)
211 		 * @private
212 		 * @param {string} str where 문자열
213 		 * @return {array} type, value 를 가지는 token collecion
214 		 * @memberOf Eco.XComp
215 		 */
216 		_tokenizeQueryWhere: function(str)
217 		{
218 			var c;				// The current character.
219 			var i = 0;			// The index of the current character.	
220 			var s;				// The string value.
221 			var n;				// The number value.
222 			var q;				// The quote character.
223 			
224 			var result = [];	// An array to hold the results.
225 			
226 			var prefix = '<>&|=*^$!';
227 			var suffix = '=>&|';
228 			
229 			var length = str.length;
230 			
231 			c = str.charAt(i);
232 			
233 			while (c) 
234 			{
235 				// Ignore whitespace.
236 				if (c <= ' ') 
237 				{
238 					i += 1;
239 					c = str.charAt(i);
240 				}
241 				// named or boolean
242 				else if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || c == '_') 
243 				{
244 					s = c;
245 					i += 1;
246 					for (;;) 
247 					{
248 						c = str.charAt(i);
249 						if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') ||
250 								(c >= '0' && c <= '9') || c === '_') {
251 							s += c;
252 							i += 1;
253 						} else {
254 							break;
255 						}
256 					}
257 					
258 					if ( s == 'true' || s == 'false' )
259 					{
260 						result.push({type:'boolean', value:s});
261 					}
262 					else
263 					{
264 						result.push({type:'named', value:s});
265 					}
266 				} 
267 				// number.
268 				// A number cannot start with a decimal point. It must start with a digit,
269 				// possibly '0'.
270 				else if (c >= '0' && c <= '9') {
271 					s = c;
272 					i += 1;
273 					// Look for more digits.
274 					for (;;) 
275 					{
276 						c = str.charAt(i);
277 						if (c < '0' || c > '9') {
278 							break;
279 						}
280 						i += 1;
281 						s += c;
282 					}
283 					// Look for a decimal fraction part.
284 					if (c === '.') {
285 						i += 1;
286 						s += c;
287 						for (;;) 
288 						{
289 							c = str.charAt(i);
290 							if (c < '0' || c > '9') {
291 								break;
292 							}
293 							i += 1;
294 							s += c;
295 						}
296 					}
297 					// Look for an exponent part.
298 					if (c === 'e' || c === 'E') {
299 						i += 1;
300 						s += c;
301 						c = str.charAt(i);
302 						if (c === '-' || c === '+') {
303 							i += 1;
304 							s += c;
305 							c = str.charAt(i);
306 						}
307 						if (c < '0' || c > '9') {
308 							trace("Bad exponent");
309 						}
310 						do {
311 							i += 1;
312 							s += c;
313 							c = str.charAt(i);
314 						} while (c >= '0' && c <= '9');
315 					}
316 					// Make sure the next character is not a letter.
317 					if (c >= 'a' && c <= 'z') 
318 					{
319 						s += c;
320 						i += 1;                
321 						trace("Bad number");
322 					}
323 					// Convert the string value to a number. If it is finite, then it is a good
324 					// token.
325 					n = +s;
326 					if (isFinite(n)) 
327 					{
328 						result.push({type:'number', value:n});
329 					}
330 					else 
331 					{
332 						trace("Bad number");
333 					}
334 				}
335 				// string
336 				else if (c === '\'' || c === '"') {
337 					s = '';
338 					q = c;
339 					i += 1;
340 					for (;;) 
341 					{
342 						c = str.charAt(i);
343 						if (c < ' ') {
344 							if (c === '\n' || c === '\r' || c === '')
345 							{
346 								trace("Unterminated string.");
347 							}
348 							else
349 							{
350 								trace("Control character in string.");
351 							}
352 						}
353 
354 						// Look for the closing quote.
355 						if (c === q) 
356 						{
357 							break;
358 						}
359 
360 						// Look for escapement.
361 						if (c === '\\') {
362 							i += 1;
363 							if (i >= length) 
364 							{
365 							   trace("Unterminated string");
366 							}
367 							c = str.charAt(i);
368 							switch (c) {
369 							case 'b':
370 								c = '\b';
371 								break;
372 							case 'f':
373 								c = '\f';
374 								break;
375 							case 'n':
376 								c = '\n';
377 								break;
378 							case 'r':
379 								c = '\r';
380 								break;
381 							case 't':
382 								c = '\t';
383 								break;
384 							case 'u':
385 								if (i >= length) 
386 								{
387 									trace("Unterminated string");
388 								}
389 								c = parseInt(str.substr(i + 1, 4), 16);
390 								if (!isFinite(c) || c < 0) {
391 									trace("Unterminated string");
392 								}
393 								c = String.fromCharCode(c);
394 								i += 4;
395 								break;
396 							}
397 						}
398 						s += c;
399 						i += 1;
400 					}
401 					i += 1;
402 					result.push({type:'string', value:s});
403 					c = str.charAt(i);
404 				} 
405 				// comment. (not use)
406 				else if (c === '/' && str.charAt(i + 1) === '/') 
407 				{
408 					i += 1;
409 					for (;;) 
410 					{
411 						c = str.charAt(i);
412 						if (c === '\n' || c === '\r' || c === '') {
413 							break;
414 						}
415 						i += 1;
416 					}
417 				} 
418 				// combining
419 				else if (prefix.indexOf(c) >= 0) 
420 				{
421 					s = c;
422 					i += 1;
423 					while (true) 
424 					{
425 						c = str.charAt(i);
426 						if (i >= length || suffix.indexOf(c) < 0) 
427 						{
428 							break;
429 						}
430 						s += c;
431 						i += 1;
432 					}
433 					result.push({type:'operator', value:s});
434 				} 
435 				// single-character operator
436 				else 
437 				{
438 					if ( "()[]".indexOf(c) > -1 )
439 					{
440 						result.push({type:'bracket', value:c});
441 					}
442 					else
443 					{
444 						result.push({type:'operator', value:c});
445 					}
446 					
447 					i += 1;            
448 					c = str.charAt(i);
449 				}	
450 			}
451 			return result;
452 		},

456 		 * @private

459 		 * @memberOf Eco.XComp

461 		_parseQueryWhere: function(where)
462 		{
463 			var pThis = Eco.XComp;
464 			var cache = pThis._parseQueryCache[where];
465 			
466 			if ( cache )
467 			{
468 				return cache;
469 			}
470 			
471 			var token, type, value, temp;
472 			var tokens = pThis._tokenizeQueryWhere(where);
473 			
474 			var depth = 0, maxDepth=0;
475 			var syntax = [];
476 			
477 			for (var i=0,len=tokens.length; i<len; i++)
478 			{
479 				token = tokens[i];
480 				type = token.type;
481 				value = token.value;
482 				//trace(type + " : " + value);
483 				if ( type == "bracket" )
484 				{
485 					if ( value == "(" )
486 					{
487 						syntax.push({depth:depth, type:"value", value:"("});
488 						depth += 1;
489 						maxDepth = depth;
490 					}
491 					else if ( value == ")" )
492 					{
493 						depth -= 1;
494 						syntax.push({depth:depth, type:"value", value:")"});						
495 					}
496 				}
497 				else if ( type == "named" )
498 				{			
499 					if ( value == "prop" || value == "curStyle" || value == "style" )
500 					{
501 						
502 						i += 1;
503 						token = tokens[i];
504 						
505 						
506 						if ( token.type == "bracket" && token.value == "[" )
507 						{
508 							i += 1;
509 							token = tokens[i];
510 							syntax.push({depth:depth, type:"operand", value:"Eco.XComp._getQueryCompValue(obj, \""+value+"\", \""+token.value+"\")"});
511 						}
512 						i += 1;
513 					}
514 					else if ( value == "typeOf" || value == "isVisible" || value == "isVisual" || value == "isEnable" )
515 					{
516 						syntax.push({depth:depth, type:"operand", value:"Eco.XComp."+value+"(obj)"});
517 					}
518 				}
519 				else if ( type == "operator" )
520 				{
521 					if ( value == "&&" || value == "||" )
522 					{
523 						syntax.push({depth:depth, type:"operator", value:value});
524 					}
525 					else
526 					{
527 						temp = "";
528 						switch (value)
529 						{
530 							case '==' :
531 								temp = "equal";
532 								break;
533 							case '!=' :
534 								temp = "notEqual";
535 								break;
536 							case '>' :
537 								temp = "greaterThan";
538 								break;
539 							case '>=' :
540 								temp = "greaterThanEqual";
541 								break;
542 							case '<' :
543 								temp = "lessThan";
544 								break;
545 							case '<=' :
546 								temp = "lessThanEqual";
547 								break;
548 							case '*=' :
549 								temp = "contains";
550 								break;	
551 							case '^=' :
552 								temp = "startWith";
553 								break;	
554 							case '$=' :
555 								temp = "endWith";
556 								break;																																				
557 						}
558 						syntax.push({depth:depth, type:"replaceOperator", value:"Eco.XComp._operators."+temp+"($1, $2)"});
559 					}
560 				} 		
561 				else if ( type == "string" )
562 				{
563 					syntax.push({depth:depth, type:"operand", value:"\""+value+"\""});
564 				}
565 				else if ( type == "number" || type == "boolean" )
566 				{		
567 					syntax.push({depth:depth, type:"operand", value:value});
568 				}
569 			}
570 
571 			var s, s0, s1;
572 			for (var i=maxDepth; i>=0; i--)
573 			{
574 				pThis._replaceOperator(i, syntax);
575 			}
576 			
577 			var result = [];
578 			for (var i=0,len=syntax.length; i<len ;i++)
579 			{
580 				s = syntax[i];
581 				result.push(syntax[i].value);
582 			}
583 			
584 			var fn = "var f = function(obj) {\n	return " + result.join(" ") + ";\n}";
585 
586 			try 
587 			{
588 				eval(fn);
589 			}
590 			catch(e)
591 			{
592 				trace("eval error:"+e.description);
593 			}
594 						
595 			pThis._parseQueryCache[where] = f;
596 			
597 			return f;

600 		/**
601 		 * where 조건문의 연산자에 피연산자를 대입한다.
602 		 * @private
603 		 * @param {number} depth 우선순위 깊이
604 		 * @param {number} syntax 변환 대상 토큰
605 		 * @memberOf Eco.XComp
606 		 */		
607 		_replaceOperator: function(depth, syntax)
608 		{
609 			var s;
610 			for (var i=0,len=syntax.length; i<len; i++)
611 			{
612 				s = syntax[i];
613 				if ( s.depth == depth )
614 				{
615 					if ( s.type == "replaceOperator" )
616 					{
617 						s0 = syntax[i-1];
618 						s1 = syntax[i+1];
619 						
620 						if ( s0 && s0.depth == depth && s0.type == "operand" && 
621 						     s1 && s1.depth == depth && s1.type == "operand" )
622 						{
623 							temp = s.value;
624 							temp = temp.replace("$1", s0.value);
625 							temp = temp.replace("$2", s1.value);
626 
627 							s.depth = s.depth-1;
628 							s.type = "operand";
629 							s.value = temp;
630 							
631 							s0.depth = null;
632 							s1.depth = null;
633 							
634 							i += 1;
635 						}
636 					}
637 				}
638 			}
639 			
640 			for (var i=(syntax.length-1); i>=0; i--)
