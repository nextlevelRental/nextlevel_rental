1 ﻿/**
  2  * @fileoverview LocalStorage와 관련된 함수.
  3  * key/value 쌍으로 data를 저장한다.
  4  */
  5  
  6 if ( !JsNamespace.exist("Eco.LocalStorage") )
  7 {
  8 	/**
  9 	 * @namespace
 10 	 * @name Eco.LocalStorage
 11 	 * @memberof! <global>
 12 	 */
 13 	JsNamespace.declare("Eco.LocalStorage", {
 14 	 
 15 
 16 	/**
 17 	 * key/value 쌍으로 data를 저장한다.<br>
 18 	 * @param {string} key The key.
 19 	 * @param {string} value The associated value for key.
 20 	 * @param {boolean=} common Whether to bring data in common area(default:false). RUNTIME Only.
 21 	 * @return {boolean} 성공시 true, 실패시 false를 반환한다.
 22 	 * @memberOf Eco.LocalStorage
 23 	 */		
 24 	setItem: function(key, value, common)
 25    	{ 
 26 		return nexacro.Application.setPrivateProfile(key, value, common);
 27 	},    
 28 
 29 	
 30 	/**
 31 	 * key에 해당하는 data를 반환한다.<br>
 32 	 * @param {string} key The key.
 33 	 * @param {boolean=} common Whether to bring data in common area(default:false). RUNTIME Only.
 34 	 * @return {string|null} key에 해당하는 값.
 35 	 * @memberOf Eco.LocalStorage
 36 	 */		
 37 	getItem: function(key, common)
 38    	{ 
 39 		return nexacro.Application.getPrivateProfile(key, common);
 40 	}
 41 	
 42 	});
 43 	
 44 	var me = Eco.LocalStorage;
 45 	//removeItem(), clear() 별도처리. 향후 엔진에 적용되면 없앨 것.
 46 	if(nexacro.Browser.toUpperCase() == "RUNTIME") {
 47 		//런타임 내부의 data 저장방식을 알 수 없어 처리 불가.
 48 		me.removeItem = nexacro._emptyFn;
 49 		me.clear = nexacro._emptyFn;
 50 		
 51 	//HTML5	
 52 	} else {
 53 	    if (nexacro.Browser == "IE" && nexacro.BrowserVersion == 6)
 54 	    {
 55 	    	me.removeItem = nexacro._emptyFn;
 56 	    	me.clear = nexacro._emptyFn;
 57 	    }
 58 	    else if (nexacro.Browser == "IE" && nexacro.BrowserVersion == 7)
 59 	    {    
 60 	    	me.removeItem = function(key) {
 61 	            var localstoragekey = application.key + application.xadl;
 62 	            var iframenode = nexacro._managerFrameNode;
 63 	            if (iframenode)
 64 	            {
 65 	                iframenode.load(localstoragekey);
 66 	                var attribute = iframenode.getAttribute(key);
 67 	                if (attribute)
 68 	                {
 69 	                	iframenode.removeAttribute(key);
 70 	                }            	
 71 	            }
 72 		    }
 73 	    	
 74 	    	me.clear = function() {
 75 	            var localstoragekey = application.key + application.xadl;
 76 	            var iframenode = nexacro._managerFrameNode;
 77 	            if (iframenode)
 78 	            {
 79 	                iframenode.load(localstoragekey);
 80 	                var attribute = iframenode.getAttribute(key);
 81 	                if (attribute)
 82 	                {
 83 	                	//http://msdn.microsoft.com/en-us/library/ms536350(v=vs.85).aspx
 84 	                	iframenode.clearAttributes();
 85 	                }            	
 86 	            }
 87 		    }	    	
 88 	    }
 89 	    else 
 90 	    {
 91 	    	/**
 92 	    	 * localStorage에서 key에 해당하는 key/value를 제거한다(RUNTIME 미지원).<br>
 93 	    	 * @param {string} key The key.
 94 	    	 * @memberOf Eco.LocalStorage
 95 	    	 */			    	
 96 	    	Eco.LocalStorage.removeItem = function(key) {
 97 	            var localstorage = window.localStorage;
 98 	            if (localstorage)
 99 	            {
100 	                var localstoragekey = application.key + application.xadl;
101 	                if (localstoragekey)
102 	                {
103 	                    var localstoragedata = localstorage.getItem(localstoragekey);
104 	                    var jsondata = {};
105 	                    
106 	                    if (localstoragedata)
107 	                    {
108 	                        jsondata = JSON.parse(localstoragedata);
109 	                    }
110 	                    
111 	                    var findkey = jsondata[key];
112 	                    if (findkey)
113 	                    {
114 	                    	delete jsondata[key];
115 	                    	
116 	                    	localstorage.setItem(localstoragekey, JSON.stringify(jsondata));
117 	                    }
118 	                }
119 	            }
120 		    	
121 		    }
122 	    	
123 	    	
124 	    	/**
125 	    	 * localStorage에서 모든 key/value를 제거한다(RUNTIME 미지원).<br>
126 	    	 * @memberOf Eco.LocalStorage
127 	    	 */			    	
128 	    	Eco.LocalStorage.clear = function() {
129 	            var localstorage = window.localStorage;
130 	            if (localstorage)
131 	            {
132 	                var localstoragekey = application.key + application.xadl;
133 	                if (localstoragekey)
134 	                {
135 	                	var jsondata = {};	
136 	                	localstorage.setItem(localstoragekey, JSON.stringify(jsondata));
137 	                }
138 	            }
139 		    	
140 		    }	 	    	
141 	    	
142 	    }
143 		
144 	}
145 
146 }
