 17 				var value = Eco.ClassUtils.peekFromList(Class, list);
 18 				Class._index++;
 19 				Class.__read = 1;
 20 				return value;
 21 			}
 22 
 23 			var proto = Class.prototype,
 24 				readIndex = Class._readIndex,
 25 				index = start || readIndex && list._index || 0;
 26 
 27 			if (!length)
 28 			{
 29 				length = list.length - index;
 30 			}
 31 
 32 			var obj = list[index];
 33 			if ( obj instanceof Class
 34 				|| options && options.readNull && obj == null && length <= 1)
 35 			{
 36 				if (readIndex)
 37 				{
 38 					list._index = index + 1;
 39 				}
 40 				//trace("readFromList==>" + list._index + "///" + obj.getClassName() + "///" + readIndex);
 41 				return obj && options && options.clone ? obj.clone() : obj;
 42 			}

 44 			obj = new Class();
 45 
 46 			if (readIndex)
 47 			{
 48 				obj.__read = true;
 49 			}

 52 				obj.__options = options;
 53 			}
 54

 56 			if ( index > 0 || length < list.length )
 57 			{
 58 				args = Array.prototype.slice.call(list, index, index + length);
 59 			}
 60 			else
 61 			{
 62 				args = list;
 63 			}
 64 			obj = obj.initialize.apply(obj, args) || obj;
 65 			if (readIndex)
 66 			{
 67 				list._index = index + obj.__read;
 68 				list.__read = obj.__read;
 69 				delete obj.__read;
 70 				if (options)
 71 				{
 72 					delete obj.__options;
 73 				}
 74 			}
 75 			return obj;
 76 		},
 77 		peekFromList: function(list, start)
 78 		{
 79 			return list[list._index = start || list._index || 0];
 80 		},
 81 		readAllFromList: function(Class, list, start, options)
 82 		{
 83 			var res = [], entry;
 84 			for (var i = start || 0, l = list.length; i < l; i++)
 85 			{
 86 				res.push(Eco.isArray(entry = list[i])
 87 					? Eco.ClassUtils.readFromList(Class, entry, 0, 0, options)
 88 					: Eco.ClassUtils.readFromList(Class, list, i, 1, options));
 89 			}
 90 			return res;
 91 		},

 93 		{
 94 			var value = Eco.ClassUtils.getNamed(list, name);
 95 			return Eco.ClassUtils.readFromList(Class, value != null ? [value] : list, start, length,
 96 					options);
 97 		},
 98 		getNamed: function(list, name)
 99 		{
100 			var arg = list[0];
101 			if (list._hasObject === undefined)
102 			{
103 				list._hasObject = list.length === 1 && Eco.isObject(arg);
104 			}
105 			if (list._hasObject)
106 			{
107 				return name ? arg[name] : arg;
108 			}
109 		},
110 		hasNamed: function(list, name)
111 		{
112 			return !!Eco.ClassUtils.getNamed(list, name);
113 		},

116 			for (var i = 0, l = arguments.length; i < l; i++)
117 			{
118 				if (arguments[i] != null)
119 				{
120 					return arguments[i];
121 				}
122 			}
123 			return null;

130 		},

138 		},

142 		},
