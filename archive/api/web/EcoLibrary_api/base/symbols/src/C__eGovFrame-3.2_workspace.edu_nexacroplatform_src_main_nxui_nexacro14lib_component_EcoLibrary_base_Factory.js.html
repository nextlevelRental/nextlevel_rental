 83 		"_getCache": function(p, poolName, type, isCreate)
 84 		{
 85 			var cache;
 86 			if ( poolName == "default" )
 87 			{
 88 				cache = p._defaultCache;
 89 				if ( isCreate && !cache )
 90 				{
 91 					p._defaultCache = {};
 92 					cache = p._defaultCache;
 93 				}
 94 				if ( !cache ) return null;
 95 				if ( !type || !type.length )
 96 				{
 97 					Eco.Logger.error({message:"poolName이 'default'면 type값이 반드시 존재해야 합니다.", stack: true});
 98 				}
 99 				var typeCache = cache[type];
100 				if ( isCreate && !typeCache )
101 				{
102 					cache[type] = [];
103 					typeCache = cache[type];
104 				}
105 				return typeCache;
106 			}
107 			else
108 			{
109 				cache = p._cache;
110 				if ( isCreate && !cache )
111 				{
112 					p._cache = {};
113 					cache = p._cache;
114 				}
115 				if ( !cache ) return null;
116 				var namedCache = cache[poolName];
117 				if ( isCreate && !namedCache )
118 				{
119 					cache[poolName] = [];
120 					namedCache = cache[poolName];
121 				}

248 					}
249 					else if ( argLens == 4 )
250 					{

264

266 			used[xcomp._id] = xcomp;
267 			return xcomp;
