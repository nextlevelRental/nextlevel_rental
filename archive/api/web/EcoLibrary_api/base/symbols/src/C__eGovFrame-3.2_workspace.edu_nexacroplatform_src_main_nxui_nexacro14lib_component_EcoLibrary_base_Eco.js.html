456 		clone: function(value, deep, exclude)
457 		{
458 			if ( Eco.isObject(value) || Eco.isArray(value) )
459 			{
460 				var objectClone,
461 					chkFunc = Eco.array.indexOf;
462 				if ( value.constructor == Array ) 
463 				{
464 					objectClone = [];
465 				}			
466 				else
467 				{
468 					objectClone = {};
469 				}
470 				
471 				for (var property in value) 
472 				{
473 					if ( value.hasOwnProperty(property) && (!exclude || chkFunc(exclude, property) == -1) )
474 					{
475 						if (!deep)
476 						{
477 							objectClone[property] = value[property]; 
478 						}
479 						else if (typeof value[property] == 'object') 
480 						{
481 							objectClone[property] = Eco.clone(value[property], deep, exclude); 
482 						}
483 						else 
484 						{
485 							objectClone[property] = value[property]; 
486 						}
487 					}
488 				}
489 				
490 				return objectClone;
491 			}
492 			else if ( Eco.isDate(value) )
493 			{
494 				var dateClone = new Date();
495 				dateClone.setTime(value.getTime());
496 				return dateClone;
497 			}
498 			else if ( Eco.isXComponent(value) )
499 			{
500 				trace("[Eco.clone] TO-DO==> clone if nexacro Component");
501 				return null;
502 			}
503 			else if ( Eco.isPrimitive(value) )
504 			{
505 				return value;
506 			}
507 			else if ( value && ( typeof value.clone == "function" ) )
508 			{
509 				return value.clone();
510 			}
511 			else if ( value && value._className && value._className.length )
512 			{
513 				var clone = new value.constructor();
514 				var values = value.getPropertiesValues(),
515 					val,
516 					resetValues = {};
517 				for (var i in values)
518 				{
519 					if ( (exclude == null ||
520 						Eco.array.indexOf(exclude, i) < 0))
521 					{
522 						val = values[i];
523 						if (deep)
524 						{
525 							resetValues[i] = Eco.clone(val, deep, exclude);
526 						}
527 						else
528 						{
529 							resetValues[i] = val;
530 						}
531 					}
532 				}
533 				clone.setProperties(resetValues);
534 				return clone;
535 			}
536 			else
537 			{
538 				return null;
539 			}
