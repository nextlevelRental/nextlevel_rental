536 		* 주어진 rect(Eco.Rectangle) 값을 다음 값으로 재구성하여 _transformCoordinates 함수를 호출하여 결과를 return한다.
537 		* rect(Eco.Rectangle) => [left, top, right, top, right, bottom, left, bottom]
538 		* @param {Eco.Rectangle} rect 4개의 지점에 대하여 transformation 처리 할 Rectangle객체
539 		* @return {array} transformation 처리 된 결과 array
540 		* @private
541 		* @memberOf Eco.Matrix

543 		"_transformCorners": function(rect)

545 			var x1 = rect.x,
546 				y1 = rect.y,

551 		},
552 		/**

554 		* 예를 들면 안쪽 변형된 사각형의 bounds는 바깥쪽 사각형이다.

600 		/**
601 		* 주어진 pt(Eco.Point) 값을 이 Matrix의 반전 값(1/Matrix)으로 transformation 적용된 지점을 얻는다.
602 		* arguments 설정은 아래 기술된 내용에 대하여 모두 처리한다.
603 		* inverseTransform(x, y) // return : Eco.Point
604 		* inverseTransform(pt) // return : Eco.Point
605 		* @param {Eco.Point} pt inverse transformation 처리 할 source 지점(x, y)
606 		* @return {Eco.Point} inverse transformation 처리 된 결과 지점(x, y)
607 		* @memberOf Eco.Matrix
608 		*/
609 		"inverseTransform": function()
610 		{
611 			if (!(pt instanceof Eco.Point))
612 			{
613 				pt = new Eco.Point(arguments[0], arguments[1]);
614 				var readargs = pt.__readArgs;
615 				return this._inverseTransform(pt, arguments[readargs]);
616 			}
617 			else
618 			{
619 				return this._inverseTransform(pt, arguments[1]);
620 			}
621 		},
622 		/**
623 		* Matrix 감도 계수(?)를 구한다.
624 		* @return {number|null} inverse 할 감도 계수 값 만약 inverse 할 수 없으면 null값
625 		* @private
626 		* @memberOf Eco.Matrix
627 		*/
628 		"_getDeterminant": function()
629 		{
630 			var det = this._a * this._d - this._b * this._c;
631 			return isFinite(det) && !MathUtil.isZero(det)
632 					&& isFinite(this._tx) && isFinite(this._ty)
633 					? det : null;
634 		},
635 		/**
636 		* 주어진 pt(Eco.Point) 값을 이 Matrix의 반전 값(1/Matrix)으로 transformation 적용된 지점을 얻는다.
637 		* @param {Eco.Point} pt inverse transformation 처리 할 source 지점(x, y)
638 		* @param {Eco.Point=} dest inverse transformation 처리 된 결과를 저장할 target 지점(x, y)
639 		* @return {Eco.Point} inverse transformation 처리 된 결과 지점(x, y)
640 		* @private
641 		* @memberOf Eco.Matrix
642 		*/
643 		"_inverseTransform": function(pt, dest)
644 		{
645 			var det = this._getDeterminant();
646 			if (!det)
647 			{
648 				return null;
649 			}
650 			var x = pt.x - this._tx,
651 				y = pt.y - this._ty;
652 			if (!dest)
653 			{
654 				dest = new Eco.Point();
655 			}
656 			return dest.set(
657 				(x * this._d - y * this._b) / det,
658 				(y * this._a - x * this._c) / det
659 			);
660 		},
661 		/**
662 		* matrix에 정의된 내용에 대하여 tranlate, scale, rotate, skew|shear 값을 생성한다.
663 		* 주의사항 : matrix을 생성하거나 메소드를 통해서 설정한 값과 일치하지 않을 수 있다.
664 		*             단지 시각적으로 보이는 형태만 설정된 값으로 처리된다.
665 		* @return {object} translation, scaling, rotation, shearing, xpSkew, xpScale, xpRotate의 속성값이 존재하는 객체
666 		* @memberOf Eco.Matrix
667 		*/
668 		"decompose": function()
669 		{
670 			var a = this._a, b = this._b, c = this._c, d = this._d;
671 			if (MathUtil.isZero(a * d - b * c))
672 			{
673 				return null;
674 			}
675 
676 			var scaleX = Math.sqrt(a * a + b * b);
677 			a /= scaleX;
678 			b /= scaleX;
679 
680 			var shear = a * c + b * d;
681 			c -= a * shear;
682 			d -= b * shear;
683 
684 			var scaleY = Math.sqrt(c * c + d * d);
685 			c /= scaleY;
686 			d /= scaleY;
687 			shear /= scaleY;
688 
689 			if (a * d < b * c)
690 			{
691 				a = -a;
692 				b = -b;
693 				shear = -shear;
694 				scaleX = -scaleX;
695 			}
696 			
697 			var res = {
698 				"translation": this.getTranslation(),
699 				"scaling": new Eco.Point(scaleX, scaleY),
700 				"rotation": -Math.atan2(b, a),
701 				"shearing": shear,
702 				"xpSkew": null,
703 				"xpScale": null,
704 				"xpRotate": null
705 			};
706 			var shearAngle = Math.atan(shear)*Eco.Matrix.RADTODEG,
707 				skewY = shearAngle - Math.atan2(this._b, this._a)*Eco.Matrix.RADTODEG,
708 				skewX = shearAngle + Math.atan2(-this._c, this._d)*Eco.Matrix.RADTODEG;
709 
710 			if ( Math.round(skewX) != (-Math.round(skewY)) )
711 			{
712 				res.xpSkew = new Eco.Point(skewX, skewY);
713 				res.xpScale = new Eco.Point(this._a, this._d);
714 			}
715 			else
716 			{
717 				res.xpScale = res.scaling;
718 				res.xpRotate = res.rotation*Eco.Matrix.RADTODEG;
719 			}
720 			return res;
721 		},
722 		/**
723 		* matrix에 정의된 모든 값을 a, c, b, d, tx, ty 값을 얻는다. 
724 		* @return {array} a, c, b, d, tx, ty 나열된 array
725 		* @memberOf Eco.Matrix
726 		*/

758 		/**
759 		* 숫자에서 방정식 a*x=b 의 x값을 해결하기 위해 x = b/a로 구하면 된다.
760 		* Matrix에서는 이와 같이 처리하기 위해 1/a에 해당하는 값을 구해야 한다.
761 		* 다시 얘기하면 숫자에서 1 = a/a;이다 여기서 1/a값에 해당되는 Matrix값을 구하는 것이 이 함수이다.
762 		* @return {Eco.Matrix} 1/this 에 해당하는 matrix 객체
763 		* @example
764 		* var m = new Eco.Matrix();
765 		* m.translate(10, 10);
766 		* trace(m.toString());	// output : result==>[[1, 0, 10], [0, 1, 10]]
767 		* var invertM = m.inverted();
768 		* trace("invertM=>" + invertM.toString());	// output : result==>invertM=>[[1, 0, -10], [0, 1, -10]]
769 		* @memberOf Eco.Matrix
770 		*/
771 		"inverted": function()
772 		{
773 			var det = this._getDeterminant();
774 			return det && new Eco.Matrix(
775 					this._d / det,
776 					-this._c / det,
777 					-this._b / det,
778 					this._a / det,
779 					(this._b * this._ty - this._d * this._tx) / det,
780 					(this._c * this._tx - this._a * this._ty) / det);
781 		},

791 		/**
792 		* matrix에 정의된 transformation 정보를 XPLATFORM transformation style값으로 구성하여 넘겨준다. 
793 		* @param {XPComp} comp XPLATFORM component
794 		* @return {String} transformation style 값
795 		* @example
796 		* var m = new Eco.Matrix();
797 		* m.translate(200, 200);
798 		* var comp = XpCompFactory.getXpComp(Div00, "Shape");
799 		* XpCompFactory.setRect(comp, 0, 0, rectWidth, rectHeight);
800 		* trace(m.getStyleString(comp));	// output : "transformation: 0,0 [translate 200,200];"
801 		* @memberOf Eco.Matrix
802 		*/
803 		"getStyleString": function(comp)
804 		{
805 			var str = "";
806 
807 			if ( !this.isIdentity() )
808 			{
809 				var transformation = this.decompose();
810 				if ( !transformation ) return str;
811 				var compRect = XpCompFactory.getRect(comp),
812 					cw = compRect.width,
813 					ch = compRect.height,
814 					cx = compRect.x,
815 					cy = compRect.y;
816 				str = "0,0 ";
817 				var translation = transformation.translation,
818 					scaling = transformation.xpScale,
819 					rotation = transformation.xpRotate,
820 					skew = transformation.xpSkew;
821 
822 				if ( scaling && ( scaling.x != 1 || scaling.y != 1) )
823 				{
824 					var sx = scaling.x,
825 						sy = scaling.y;
826 					if ( sx != 1 )
827 					{
828 						if ( sx < 0 )
829 						{
830 							translation.x += -cx + cx*sx;
831 							cw = (-1*sx*cw);
832 						}
833 						else
834 						{
835 							translation.x += -cx + cx*sx;
836 							cw = (sx*cw);
837 						}
838 					}
839 					if ( sy != 1 )
840 					{
841 						if ( sy < 0 )
842 						{
843 							translation.y += -cy + cy*sy;
844 							ch = (-1*sy*ch);
845 						}
846 						else
847 						{
848 							translation.y += -cy + cy*sy;
849 							ch = (sy*ch);
850 						}
851 					}
852 					str += "[scale " + sx + "," + sy +"]";
853 				}
854 				if ( skew )
855 				{
856 					var radY = Math.tan(Math.atan2(this._c, this._a)),
857 						radX = Math.tan(Math.atan2(this._b, this._d));
858 
859 					//trace("radX==>" + Math.atan2(this._b, this._d));
860 					//trace("x, y half gap==>" + (ch*radX*0.5) + "," + (cw*radY*0.5));
861 					//trace("x, y gap==>" + (ch*radX) + "," + (cw*radY));
862 					//trace("x, y angle==>" + (radX*180/Math.PI) + "," + (radY*180/Math.PI));
863 					//trace("x, y radian==>" + radX + "," + radY);
864 					translation.x -= ch*radX*0.5;
865 					translation.y -= cw*radY*0.5;
866 					str += "[skew " + skew.x + "," + skew.y +"]";
867 				}
868 				if ( rotation )
869 				{
870 					rotation = rotation;
871 					if ( Math.abs(rotation) != 360 || Math.abs(rotation) != 180 )
872 					{
873 						var cx = compRect.x,
874 							cy = compRect.y;
875 						var pt = new Eco.Point(cx, cy);
876 						pt = pt.rotate(rotation, 0, 0);
877 						translation.x += (pt.x - cx); //pt.x;
878 						translation.y += (pt.y - cy); //pt.y;
879 						str += "[rotate " + (rotation) +"]";
880 					}
881 				}
882 
883 
884 				var tx = translation.x, 
885 					ty = translation.y;
886 				if ( tx != 0 || ty != 0  )
887 				{
888 					str += "[translate " + tx + "," + ty +"]";
889 				}
890 
891 				str = "transformation: " + str + ";";
892 			}
893 			//trace(str);
894 			return str;
895 		}
