623 		* @param {object} object 대상 객체

718 		* // 	</Dataset>	
719 		* @memberOf Eco.object
720 		*/
721 		toDataset: function(object, ds, dir)
722 		{
723 			var p, i, len,
724 				value;
725 			
726 			if (dir == "vert")
727 			{
728 				if (!Eco.isEmpty(object)) 
729 				{
730 					ds.addColumn("name", "string", 256);
731 					ds.addColumn("value", "string", 256);
732 				}
733 				
734 				for (p in object)
735 				{
736 					if ( object.hasOwnProperty(p) )
737 					{
738 						value = object[p] || "";
739 						
740 						len = ds.addRow();
741 						ds.setColumn(len, "name", p);
742 						ds.setColumn(len, "value", value);
743 					}
744 				}
745 			}
746 			else 
747 			{
748 				if (!Eco.isEmpty(object)) 
749 				{
750 					for (p in object)
751 					{
752 						if ( object.hasOwnProperty(p) )
753 						{
754 							ds.addColumn(p, "string", 256);
755 						}
756 					}
757 					
758 					len = ds.addRow();
759 					
760 					for (p in object)
761 					{
762 						if ( object.hasOwnProperty(p) )
763 						{
764 							value = object[p] || "";
765 							ds.setColumn(len, p, value);
766 						}
767 					}
768 				}
769 			}
770 		},

772 		/**
773 		* object에 속성들(name, value) 를 dataset로 부터 구성한다.<br>
774 		* dir 값이 vert 이면 dataset에 name Column, value Column 필드명에서<br>
775 		* record별로 정보을 얻은 후 object 속성들로 구성하는 방법<br>
776 		* dir 값이 horz 이면 dataset의 Column들의 id를 속성 명칭,<br>
777 		* Column에 0번째 row 값을 value로 object 속성으로 구성된다.
778 		* @param {object} object Dataset값을 참조해서 새로 구성되는 객체
779 		* @param {dataset} ds 객체구성시 참조되는 Dataset
780 		* @param {string=} dir default horz(horz/vert)
781 		* @example
782 		* var info = {};
783 		* Eco.object.fromDataset(info, dsVert, "vert");
784 		* for (var p in info)
785 		* {
786 		* 	trace(p + ":" + info[p]);
787 		* 	// output : "string": "가",
788 		* 	// output : "int": 10,
789 		* 	// output : "float": 0.8,
790 		* 	// output : "bool": true
791 		* }
792 		* @memberOf Eco.object
793 		*/
794 		fromDataset: function(object, ds, dir)
795 		{
796 			var i, len,
797 				name, value,
798 				isEmpty = Eco.isEmpty;
799 			
800 			if (dir == "vert")
801 			{
802 				for(i = 0, len = ds.getRowCount(); i < len; i++)
803 				{
804 					name = ds.getColumn(i, "name");
805 					if (isEmpty(name)) continue;
806 					
807 					value = ds.getColumn(i, "value") || "";
808 					object[name] = value;
809 				}
810 			}
811 			else
812 			{
813 				for(i = 0, len = ds.getColCount(); i < len; i++)
814 				{
815 					name = ds.getColID(i);
816 					if (isEmpty(name)) continue;
817 					
818 					value = ds.getColumn(0, name) || "";
819 					object[name] = value;
820 				}
821 			}
822 		}
823
